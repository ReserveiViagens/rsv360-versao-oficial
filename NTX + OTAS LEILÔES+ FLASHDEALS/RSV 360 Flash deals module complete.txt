  id SERIAL PRIMARY KEY,
  wishlist_id INTEGER REFERENCES shared_wishlists(id) ON DELETE CASCADE,
  user_id INTEGER REFERENCES customers(id) ON DELETE CASCADE,
  role VARCHAR(20) DEFAULT 'member', -- 'admin', 'member'
  joined_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(wishlist_id, user_id)
);

CREATE INDEX idx_wishlist_members_wishlist ON wishlist_members(wishlist_id);
CREATE INDEX idx_wishlist_members_user ON wishlist_members(user_id);

-- Tabela de Items da Wishlist
CREATE TABLE wishlist_items (
  id SERIAL PRIMARY KEY,
  wishlist_id INTEGER REFERENCES shared_wishlists(id) ON DELETE CASCADE,
  property_id INTEGER REFERENCES content(id) ON DELETE CASCADE,
  added_by INTEGER REFERENCES customers(id) ON DELETE SET NULL,
  votes_up INTEGER DEFAULT 0,
  votes_down INTEGER DEFAULT 0,
  added_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_wishlist_items_wishlist ON wishlist_items(wishlist_id);
CREATE INDEX idx_wishlist_items_property ON wishlist_items(property_id);

-- Tabela de Votos

  id SERIAL PRIMARY KEY,
  user_id INTEGER REFERENCES customers(id) ON DELETE CASCADE,
  item_id INTEGER REFERENCES wishlist_items(id) ON DELETE CASCADE,
  vote VARCHAR(10) CHECK (vote IN ('up', 'down', 'maybe')),
  timestamp TIMESTAMP DEFAULT NOW(),
  UNIQUE(user_id, item_id)
);

CREATE INDEX idx_votes_item ON votes(item_id);

-- Tabela de Split Payment
CREATE TABLE payment_splits (
  id SERIAL PRIMARY KEY,
  booking_id INTEGER REFERENCES bookings(id) ON DELETE CASCADE,

  paid_at TIMESTAMP,
  payment_method VARCHAR(50),
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_payment_splits_booking ON payment_splits(booking_id);

-- Função para atualizar votos automaticamente
CREATE OR REPLACE FUNCTION update_vote_counts()
RETURNS TRIGGER AS $$
BEGIN
  IF TG_OP = 'INSERT' OR TG_OP = 'UPDATE' THEN
    UPDATE wishlist_items
    SET 
      votes_up = (SELECT COUNT(*) FROM votes WHERE item_id = NEW.item_id AND vote = 'up'),
      votes_down = (SELECT COUNT(*) FROM votes WHERE item_id = NEW.item_id AND vote = 'down')
    WHERE id = NEW.item_id;
  ELSIF TG_OP = 'DELETE' THEN
    UPDATE wishlist_items
    SET 
      votes_up = (SELECT COUNT(*) FROM votes WHERE item_id = OLD.item_id AND vote = 'up'),
      votes_down = (SELECT COUNT(*) FROM votes WHERE item_id = OLD.item_id AND vote = 'down')
    WHERE id = OLD.item_id;
  END IF;
  RETURN NULL;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER vote_count_trigger
AFTER INSERT OR UPDATE OR DELETE ON votes
FOR EACH ROW EXECUTE FUNCTION update_vote_counts();

Criar backend/src/group-travel/services/wishlist.service.ts:
typescriptimport { pool } from '@config/database';
import { redis } from '@config/database';
import {
  SharedWishlist,
  CreateWishlistDTO,
  AddItemDTO,
  WishlistItem,
  WishlistMember,
} from '../types';

export class WishlistService {
  private readonly CACHE_TTL = 3600; // 1 hora

  /**
   * Criar nova wishlist compartilhada
   */
  async createWishlist(
    userId: number,
    data: CreateWishlistDTO
  ): Promise<SharedWishlist> {
    const client = await pool.connect();
    
    try {
      await client.query('BEGIN');

      // Criar wishlist
      const wishlistResult = await client.query(
        `INSERT INTO shared_wishlists (name, created_by)
         VALUES ($1, $2)
         RETURNING *`,
        [data.name, userId]
      );

      const wishlist = wishlistResult.rows[0];

      // Adicionar criador como admin
      await client.query(
        `INSERT INTO wishlist_members (wishlist_id, user_id, role)
         VALUES ($1, $2, 'admin')`,
        [wishlist.id, userId]
      );

      // Convidar membros se fornecidos
      if (data.member_emails && data.member_emails.length > 0) {
        for (const email of data.member_emails) {
          // Buscar user_id pelo email
          const userResult = await client.query(
            'SELECT id FROM customers WHERE email = $1',
            [email]
          );

          if (userResult.rows.length > 0) {
            await client.query(
              `INSERT INTO wishlist_members (wishlist_id, user_id, role)
               VALUES ($1, $2, 'member')
               ON CONFLICT (wishlist_id, user_id) DO NOTHING`,
              [wishlist.id, userResult.rows[0].id]
            );
          }
        }
      }

      await client.query('COMMIT');

      // Invalidar cache
      await this.invalidateWishlistCache(wishlist.id);

      return wishlist;
    } catch (error) {
      await client.query('ROLLBACK');
      throw error;
    } finally {
      client.release();
    }
  }

  /**
   * Buscar wishlists do usuário
   */
  async getUserWishlists(userId: number): Promise<SharedWishlist[]> {
    const cacheKey = `wishlists:user:${userId}`;
    
    // Tentar cache
    const cached = await redis.get(cacheKey);
    if (cached) {
      return JSON.parse(cached);
    }

    const result = await pool.query(
      `SELECT sw.*, 
              COUNT(DISTINCT wi.id) as item_count,
              COUNT(DISTINCT wm.user_id) as member_count
       FROM shared_wishlists sw
       INNER JOIN wishlist_members wm ON sw.id = wm.wishlist_id
       LEFT JOIN wishlist_items wi ON sw.id = wi.wishlist_id
       WHERE wm.user_id = $1
       GROUP BY sw.id
       ORDER BY sw.updated_at DESC`,
      [userId]
    );

    const wishlists = result.rows;

    // Cachear por 1 hora
    await redis.setex(cacheKey, this.CACHE_TTL, JSON.stringify(wishlists));

    return wishlists;
  }

  /**
   * Buscar wishlist por ID com membros e items
   */
  async getWishlistById(
    wishlistId: number,
    userId: number
  ): Promise<SharedWishlist | null> {
    const cacheKey = `wishlist:${wishlistId}`;
    
    // Tentar cache
    const cached = await redis.get(cacheKey);
    if (cached) {
      return JSON.parse(cached);
    }

    // Verificar se user tem acesso
    const accessCheck = await pool.query(
      'SELECT 1 FROM wishlist_members WHERE wishlist_id = $1 AND user_id = $2',
      [wishlistId, userId]
    );

    if (accessCheck.rows.length === 0) {
      throw new Error('Access denied');
    }

    // Buscar wishlist
    const wishlistResult = await pool.query(
      'SELECT * FROM shared_wishlists WHERE id = $1',
      [wishlistId]
    );

    if (wishlistResult.rows.length === 0) {
      return null;
    }

    const wishlist = wishlistResult.rows[0];

    // Buscar membros
    const membersResult = await pool.query(
      `SELECT wm.*, c.name, c.email
       FROM wishlist_members wm
       INNER JOIN customers c ON wm.user_id = c.id
       WHERE wm.wishlist_id = $1`,
      [wishlistId]
    );

    wishlist.members = membersResult.rows;

    // Buscar items com votos
    const itemsResult = await pool.query(
      `SELECT wi.*, 
              c.name as property_name,
              c.base_price as property_price,
              COALESCE(v.vote, NULL) as user_vote
       FROM wishlist_items wi
       INNER JOIN content c ON wi.property_id = c.id
       LEFT JOIN votes v ON wi.id = v.item_id AND v.user_id = $2
       WHERE wi.wishlist_id = $1
       ORDER BY (wi.votes_up - wi.votes_down) DESC, wi.added_at DESC`,
      [wishlistId, userId]
    );

    wishlist.items = itemsResult.rows;

    // Cachear
    await redis.setex(cacheKey, this.CACHE_TTL, JSON.stringify(wishlist));

    return wishlist;
  }

  /**
   * Adicionar propriedade à wishlist
   */
  async addItem(userId: number, data: AddItemDTO): Promise<WishlistItem> {
    // Verificar acesso
    const accessCheck = await pool.query(
      'SELECT 1 FROM wishlist_members WHERE wishlist_id = $1 AND user_id = $2',
      [data.wishlist_id, userId]
    );

    if (accessCheck.rows.length === 0) {
      throw new Error('Access denied');
    }

    // Verificar se já existe
    const existsCheck = await pool.query(
      'SELECT id FROM wishlist_items WHERE wishlist_id = $1 AND property_id = $2',
      [data.wishlist_id, data.property_id]
    );

    if (existsCheck.rows.length > 0) {
      throw new Error('Property already in wishlist');
    }

    const result = await pool.query(
      `INSERT INTO wishlist_items (wishlist_id, property_id, added_by)
       VALUES ($1, $2, $3)
       RETURNING *`,
      [data.wishlist_id, data.property_id, userId]
    );

    // Atualizar timestamp da wishlist
    await pool.query(
      'UPDATE shared_wishlists SET updated_at = NOW() WHERE id = $1',
      [data.wishlist_id]
    );

    // Invalidar cache
    await this.invalidateWishlistCache(data.wishlist_id);

    return result.rows[0];
  }

  /**
   * Remover item da wishlist
   */
  async removeItem(
    userId: number,
    wishlistId: number,
    itemId: number
  ): Promise<void> {
    // Verificar se é admin ou quem adicionou
    const checkResult = await pool.query(
      `SELECT wi.added_by, wm.role
       FROM wishlist_items wi
       INNER JOIN wishlist_members wm ON wi.wishlist_id = wm.wishlist_id
       WHERE wi.id = $1 AND wi.wishlist_id = $2 AND wm.user_id = $3`,
      [itemId, wishlistId, userId]
    );

    if (checkResult.rows.length === 0) {
      throw new Error('Access denied');
    }

    const { added_by, role } = checkResult.rows[0];

    if (added_by !== userId && role !== 'admin') {
      throw new Error('Only admins or the user who added can remove');
    }

    await pool.query('DELETE FROM wishlist_items WHERE id = $1', [itemId]);

    // Invalidar cache
    await this.invalidateWishlistCache(wishlistId);
  }

  /**
   * Convidar membro
   */
  async inviteMember(
    userId: number,
    wishlistId: number,
    email: string
  ): Promise<void> {
    // Verificar se é admin
    const adminCheck = await pool.query(
      'SELECT 1 FROM wishlist_members WHERE wishlist_id = $1 AND user_id = $2 AND role = $3',
      [wishlistId, userId, 'admin']
    );

    if (adminCheck.rows.length === 0) {
      throw new Error('Only admins can invite members');
    }

    // Buscar user pelo email
    const userResult = await pool.query(
      'SELECT id FROM customers WHERE email = $1',
      [email]
    );

    if (userResult.rows.length === 0) {
      throw new Error('User not found');
    }

    const invitedUserId = userResult.rows[0].id;

    // Adicionar membro
    await pool.query(
      `INSERT INTO wishlist_members (wishlist_id, user_id, role)
       VALUES ($1, $2, 'member')
       ON CONFLICT (wishlist_id, user_id) DO NOTHING`,
      [wishlistId, invitedUserId]
    );

    // Invalidar cache
    await this.invalidateWishlistCache(wishlistId);
  }

  /**
   * Remover membro
   */
  async removeMember(
    userId: number,
    wishlistId: number,
    memberId: number
  ): Promise<void> {
    // Verificar se é admin
    const adminCheck = await pool.query(
      'SELECT 1 FROM wishlist_members WHERE wishlist_id = $1 AND user_id = $2 AND role = $3',
      [wishlistId, userId, 'admin']
    );

    if (adminCheck.rows.length === 0) {
      throw new Error('Only admins can remove members');
    }

    // Não pode remover o criador
    const creatorCheck = await pool.query(
      'SELECT created_by FROM shared_wishlists WHERE id = $1',
      [wishlistId]
    );

    if (creatorCheck.rows[0].created_by === memberId) {
      throw new Error('Cannot remove creator');
    }

    await pool.query(
      'DELETE FROM wishlist_members WHERE wishlist_id = $1 AND user_id = $2',
      [wishlistId, memberId]
    );

    // Invalidar cache
    await this.invalidateWishlistCache(wishlistId);
  }

  /**
   * Invalidar cache de wishlist
   */
  private async invalidateWishlistCache(wishlistId: number): Promise<void> {
    const keys = await redis.keys(`wishlist:${wishlistId}*`);
    if (keys.length > 0) {
      await redis.del(...keys);
    }

    // Invalidar cache de usuários membros
    const members = await pool.query(
      'SELECT user_id FROM wishlist_members WHERE wishlist_id = $1',
      [wishlistId]
    );

    for (const member of members.rows) {
      await redis.del(`wishlists:user:${member.user_id}`);
    }
  }
}

export default new WishlistService();

      }

      // Inserir ou atualizar voto

        `INSERT INTO votes (user_id, item_id, vote, timestamp)
         VALUES ($1, $2, $3, NOW())
         ON CONFLICT (user_id, item_id)
         DO UPDATE SET vote = $3, timestamp = NOW()
         RETURNING *`,
        [userId, data.item_id, data.vote]
      );

      await client.query('COMMIT');

      // Invalidar cache do item

      );

      }

      return result.rows[0];
    } catch (error) {

      throw error;
    } finally {
      client.release();
    }
  }

function parseFields(fieldsString) {
  return fieldsString.split(',').map(field => {

async function promptForFields() {
  const fields = [];

import { redis } from '@config/database';
import { {{pascalCase name}}, Create{{pascalCase name}}DTO, Update{{pascalCase name}}DTO } from '../types';

export class {{pascalCase name}}Service {
  private readonly CACHE_TTL = 3600;

  /**
   * Criar {{lowerCase name}}
   */
  async create(data: Create{{pascalCase name}}DTO): Promise<{{pascalCase name}}> {
    const result = await pool.query(
      `INSERT INTO {{snakeCase pluralName}} ({{#each fields}}{{snakeCase this.name}}{{#unless @last}}, {{/unless}}{{/each}})
       VALUES ({{#each fields}}${{inc @index}}{{#unless @last}}, {{/unless}}{{/each}})
       RETURNING *`,
      [{{#each fields}}data.{{camelCase this.name}}{{#unless @last}}, {{/unless}}{{/each}}]
    );

    // Invalidar cache
    await this.invalidateCache();

    return result.rows[0];
  }

  /**
   * Buscar por ID
   */
  async findById(id: number): Promise<{{pascalCase name}} | null> {
    const cacheKey = `{{snakeCase name}}:${id}`;
    
    // Tentar cache
    const cached = await redis.get(cacheKey);
    if (cached) {
      return JSON.parse(cached);
    }

    const result = await pool.query(
      'SELECT * FROM {{snakeCase pluralName}} WHERE id = $1',
      [id]
    );

    if (result.rows.length === 0) {
      return null;
    }

    const {{camelCase name}} = result.rows[0];

    // Cachear
    await redis.setex(cacheKey, this.CACHE_TTL, JSON.stringify({{camelCase name}}));

    return {{camelCase name}};
  }

  /**
   * Listar todos
   */
  async findAll(page: number = 1, limit: number = 20): Promise<{
    data: {{pascalCase name}}[];
    total: number;
    page: number;
    pages: number;
  }> {
    const offset = (page - 1) * limit;

    const [dataResult, countResult] = await Promise.all([
      pool.query(
        'SELECT * FROM {{snakeCase pluralName}} ORDER BY created_at DESC LIMIT $1 OFFSET $2',
        [limit, offset]
      ),
      pool.query('SELECT COUNT(*) FROM {{snakeCase pluralName}}'),
    ]);

    const total = parseInt(countResult.rows[0].count);
    const pages = Math.ceil(total / limit);

    return {
      data: dataResult.rows,
      total,
      page,
      pages,
    };
  }

  /**
   * Atualizar
   */

markdown# 🎯 DEBUGGING MASTER FRAMEWORK
Você é um Desenvolvedor Pleno Sênior especializado em análise profunda e resolução sistemática de problemas de código.

## 🧠 METODOLOGIA DE RACIOCÍNIO

Use SEMPRE as seguintes técnicas combinadas:

### 1️⃣ CoT (Chain of Thought) - Cadeia de Pensamento
- Explique CADA passo do seu raciocínio
- Mostre o processo lógico completo
- Nunca pule etapas mentais

### 2️⃣ ToT (Tree of Thought) - Árvore de Pensamento
- Explore MÚLTIPLAS hipóteses simultaneamente
- Ramifique possibilidades de causa raiz
- Avalie cada ramo antes de decidir o caminho

### 3️⃣ SoT (System of Thought) - Sistema de Pensamento
- Considere o sistema INTEIRO, não só o código isolado
- Analise dependências, integrações, ambiente
- Pense em impactos colaterais

## 🔬 PROCESSO OBRIGATÓRIO DE 8 FASES

### FASE 1: COMPREENSÃO PROFUNDA 🎯
**Objetivo:** Entender o problema completamente antes de qualquer ação

**Ações:**
```
1. Ler a mensagem de erro COMPLETA (stack trace inteiro)
2. Identificar:
   - Arquivo exato onde erro ocorre
   - Linha exata do erro
   - Tipo do erro (syntax, runtime, logic, type, etc)
   - Contexto de execução (ambiente, estado da aplicação)
3. Reproduzir mentalmente o fluxo de execução até o erro
```

**Output:**
```
📍 ANÁLISE INICIAL:
- Erro: [tipo exato]
- Local: [arquivo:linha]
- Contexto: [quando ocorre]
- Mensagem: [mensagem completa]
```

---

### FASE 2: INVESTIGAÇÃO DO CONTEXTO 🔍
**Objetivo:** Coletar TODOS os dados relevantes

**Ações:**
```
1. Examinar código em volta (10 linhas antes/depois)
2. Identificar variáveis envolvidas e seus tipos
3. Verificar imports/dependências relacionadas
4. Checar configurações (env, tsconfig, package.json)
5. Analisar logs disponíveis
6. Verificar histórico de commits recentes relacionados
```

**Output:**
```
📊 CONTEXTO COLETADO:
- Variáveis envolvidas: [lista com tipos]
- Dependências: [lista de imports/packages]
- Configuração: [configs relevantes]
- Histórico: [mudanças recentes]
```

---

### FASE 3: GERAÇÃO DE HIPÓTESES (ToT) 🌳
**Objetivo:** Criar TODAS as possíveis causas raiz

**Ações:**
```
1. Listar TODAS as causas possíveis (mínimo 5)
2. Para cada causa, identificar:
   - Probabilidade (Alta/Média/Baixa)
   - Evidências a favor
   - Evidências contra
   - Impacto se for a causa real
```

**Output:**
```
🌲 ÁRVORE DE HIPÓTESES:

Hipótese 1: [descrição]
├─ Probabilidade: [Alta/Média/Baixa]
├─ Evidências +: [lista]
├─ Evidências -: [lista]
└─ Impacto: [Alto/Médio/Baixo]

Hipótese 2: [descrição]
├─ Probabilidade: [Alta/Média/Baixa]
├─ Evidências +: [lista]
├─ Evidências -: [lista]
└─ Impacto: [Alto/Médio/Baixo]

[... pelo menos 5 hipóteses]

🎯 HIPÓTESE PRINCIPAL: [escolha justificada]
```

---

### FASE 4: ANÁLISE DE CAUSA RAIZ 🎯
**Objetivo:** Identificar a VERDADEIRA causa, não só o sintoma

**Ações:**
```
1. Aplicar os "5 Porquês":
   - Por que erro X ocorreu? [resposta 1]
   - Por que [resposta 1]? [resposta 2]
   - Por que [resposta 2]? [resposta 3]
   - Por que [resposta 3]? [resposta 4]
   - Por que [resposta 4]? [CAUSA RAIZ]

2. Identificar se é:
   - Erro de lógica de negócio
   - Erro de tipo/tipagem
   - Erro de estado/lifecycle
   - Erro de configuração
   - Erro de dependência/versão
   - Erro de ambiente
   - Erro de arquitetura
```

**Output:**
```
🔬 CAUSA RAIZ IDENTIFICADA:
- Categoria: [tipo de erro]
- Causa primária: [descrição detalhada]
- Componente responsável: [arquivo/módulo exato]
- Linha específica: [número da linha]
- Razão técnica: [explicação profunda]

🔍 OS 5 PORQUÊS:
1. Por quê? [resposta]
2. Por quê? [resposta]
3. Por quê? [resposta]
4. Por quê? [resposta]
5. CAUSA RAIZ: [resposta final]
```

---

### FASE 5: DESIGN DA SOLUÇÃO 🎨
**Objetivo:** Planejar a correção ANTES de escrever código

**Ações:**
```
1. Desenhar mentalmente a solução
2. Listar TODOS os arquivos que serão alterados
3. Para cada alteração:
   - O que será mudado
   - Por que será mudado
   - Impactos da mudança
4. Considerar alternativas (mínimo 2)
5. Escolher melhor abordagem justificando
```
