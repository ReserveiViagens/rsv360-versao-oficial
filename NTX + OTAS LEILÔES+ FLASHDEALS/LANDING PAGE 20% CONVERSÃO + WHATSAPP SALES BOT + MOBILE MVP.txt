// whatsapp-sales-bot.js
const express = require('express');
const axios = require('axios');
require('dotenv').config();

const app = express();
app.use(express.json());

const PHONE_NUMBER_ID = process.env.PHONE_NUMBER_ID;
const ACCESS_TOKEN = process.env.WHATSAPP_TOKEN;
const VERSION = 'v19.0';

async function sendWhatsApp(to, message) {
  const url = `https://graph.facebook.com/${VERSION}/${PHONE_NUMBER_ID}/messages`;
  
  return axios.post(url, {
    messaging_product: 'whatsapp',
    to: to,
    type: 'template',
    template: { name: 'rsv360_sales_flow', language: { code: 'pt_BR' }, components: message }
  }, {
    headers: {
      Authorization: `Bearer ${ACCESS_TOKEN}`,
      'Content-Type': 'application/json'
    }
  });
}

// Webhook WhatsApp
app.post('/webhook', async (req, res) => {
  const entry = req.body.entry[0].changes[0];
  const phone = entry.value.contacts[0].wa_id;
  const message = entry.value.messages[0].text.body.toLowerCase();
  
  if (message.includes('caldas novas') || message.includes('hotel')) {
    await sendWhatsApp(phone, {
      type: 'body',
      parameters: [{ type: 'text', text: 'Hotel identificado!' }]
    });
    
    // Criação automática tenant
    await createHotelTenant(phone);
  }
  
  res.sendStatus(200);
});

async function createHotelTenant(phone) {
  // Integração RSV360 Admin API
  await axios.post('https://admin.rsv360.com/api/tenants/quick', {
    ownerPhone: phone,
    region: 'caldas_novas',
    autoLeiloes: true
  });
}

app.listen(3002, () => console.log('🤖 WhatsApp Sales Bot ativo'));

// sales-funnel-integration.js
const { PrismaClient } = require('@prisma/client');
const prisma = new PrismaClient();

class SalesFunnel {
  
  async handleInboundWhatsapp(message) {
    const { from, body } = message;
    
    // 1. Lead scoring + Stage
    const leadStage = this.scoreLead(body);
    const lead = await this.upsertLead(from, leadStage);
    
    // 2. Trigger próximo passo automação
    await this.triggerNextStep(lead);
    
    // 3. Analytics → Amplitude/Mixpanel
    await this.trackConversion(lead, leadStage);
  }

  scoreLead(message) {
    if (message.includes('caldas novas') || message.includes('hotel')) return 'HOT_LEAD';
    if (message.includes('interessado') || message.includes('demo')) return 'WARM_LEAD';
    return 'COLD_LEAD';
  }

  async triggerNextStep(lead) {
    const templates = {
      HOT_LEAD: 'rsv360_welcome',
      WARM_LEAD: 'leilao_resultados', 
      COLD_LEAD: 'caldas_novas_case'
    };

    await sendWhatsApp(lead.phone, templates[lead.stage]);
  }
}
