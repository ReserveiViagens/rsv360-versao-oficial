// modules/ota/adapters/base.ts
export interface OTAReservation {
  reservationId: string
  hotelId: string
  guestName: string
  roomType: string
  checkIn: string
  checkOut: string
  rate: number
  status: 'CONFIRMED' | 'CANCELLED' | 'MODIFIED'
  source: string
}

export abstract class OTAAdapter {
  abstract pullReservations(hotelId: string, from: Date, to: Date): Promise<OTAReservation[]>
  abstract pushReservation(reservation: OTAReservation): Promise<{ success: boolean; externalId: string }>
  abstract cancelReservation(reservationId: string): Promise<boolean>
}

export class BookingAdapter extends OTAAdapter {
  constructor(private apiKey: string, private hotelId: string) {
    super()
  }

  async pullReservations(from: Date, to: Date): Promise<OTAReservation[]> {
    const response = await fetch(
      `https://supply-xml.booking.com/hotels/xml/feed?hotel_id=${this.hotelId}&from=${from.toISOString()}&to=${to.toISOString()}`,
      {
        headers: {
          'Authorization': `Bearer ${this.apiKey}`,
          'Content-Type': 'application/json'
        }
      }
    )

    if (!response.ok) throw new Error(`Booking.com API error: ${response.status}`)
    
    const data = await response.json()
    return data.reservations || []
  }

  async pushReservation(reservation: OTAReservation): Promise<{ success: boolean; externalId: string }> {
    const response = await fetch(`https://supply-xml.booking.com/hotels/xml/reservation`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        hotelId: this.hotelId,
        reservationId: reservation.reservationId,
        guestName: reservation.guestName,
        roomType: reservation.roomType,
        checkIn: reservation.checkIn,
        checkOut: reservation.checkOut,
        rate: reservation.rate
      })
    })

    return {
      success: response.ok,
      externalId: response.ok ? await response.text() : ''
    }
  }

  async cancelReservation(reservationId: string): Promise<boolean> {
    const response = await fetch(
      `https://supply-xml.booking.com/hotels/xml/reservation/${reservationId}/cancel`,
      {
        method: 'DELETE',
        headers: { 'Authorization': `Bearer ${this.apiKey}` }
      }
    )
    return response.ok
  }
}

export class ExpediaAdapter extends OTAAdapter {
  constructor(private apiKey: string, private hotelId: string) {
    super()
  }

  async pullReservations(from: Date, to: Date): Promise<OTAReservation[]> {
    const response = await fetch(
      `https://partner.expedia.com/api/reservations?hotelId=${this.hotelId}&from=${from.toISOString()}&to=${to.toISOString()}`,
      {
        headers: {
          'Authorization': `Bearer ${this.apiKey}`,
          'Content-Type': 'application/json'
        }
      }
    )

    const data = await response.json()
    return data.reservations?.map((r: any) => ({
      reservationId: r.reservationId,
      hotelId: this.hotelId,
      guestName: r.guest?.name || 'Unknown',
      roomType: r.rooms[0]?.roomType || 'STANDARD',
      checkIn: r.checkIn,
      checkOut: r.checkOut,
      rate: r.rate,
      status: r.status,
      source: 'expedia'
    })) || []
  }

  async pushReservation(reservation: OTAReservation): Promise<{ success: boolean; externalId: string }> {
    const response = await fetch('https://partner.expedia.com/api/reservations', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(reservation)
    })

    return {
      success: response.ok,
      externalId: response.ok ? await response.text() : ''
    }
  }

  async cancelReservation(reservationId: string): Promise<boolean> {
    const response = await fetch(
      `https://partner.expedia.com/api/reservations/${reservationId}`,
      {
        method: 'DELETE',
        headers: { 'Authorization': `Bearer ${this.apiKey}` }
      }
    )
    return response.ok
  }
}
2. OTA Service (Orquestrador)

// services/ota-service.ts
import { BookingAdapter, ExpediaAdapter, OTAAdapter, OTAReservation } from '@/modules/ota/adapters/base'
import { prisma } from '@/lib/prisma'

export class OTAService {
  private adapters: Map<string, OTAAdapter> = new Map()

  constructor() {
    // Inicializar adapters com secrets do banco
    this.initializeAdapters()
  }

  private async initializeAdapters() {
    const configs = await prisma.otaConfig.findMany()
    
    for (const config of configs) {
      let adapter: OTAAdapter
      
      switch (config.ota) {
        case 'booking_com':
          adapter = new BookingAdapter(config.apiKey, config.hotelId)
          break
        case 'expedia':
          adapter = new ExpediaAdapter(config.apiKey, config.hotelId)
          break
        default:
          continue
      }
      
      this.adapters.set(`${config.hotelId}_${config.ota}`, adapter)
    }
  }

  async syncReservations(hotelId: string, from: Date, to: Date): Promise<OTAReservation[]> {
    const allReservations: OTAReservation[] = []

    // Executar sync paralelo por OTA
    const syncPromises = Array.from(this.adapters.entries())
      .filter(([key]) => key.startsWith(hotelId))
      .map(async ([key, adapter]) => {
        try {
          const reservations = await adapter.pullReservations(from, to)
          allReservations.push(...reservations.map(r => ({ ...r, source: key.split('_')[1] })))
        } catch (error) {
          console.error(`OTA Sync failed for ${key}:`, error)
        }
      })

    await Promise.allSettled(syncPromises)
    return allReservations
  }

  async pushReservation(reservation: OTAReservation): Promise<{ success: boolean }> {
    const adapterKey = `${reservation.hotelId}_${reservation.source}`
    const adapter = this.adapters.get(adapterKey)
    
    if (!adapter) {
      throw new Error(`No adapter found for ${reservation.source}`)
    }

    const result = await adapter.pushReservation(reservation)
    return { success: result.success }
  }
}

            />
            <span className="text-sm">Auto-refresh (5s)</span>
          </label>
        </div>
      </div>

      <Row gutter={[24, 24]}>
        {otaStatus?.map((status: OTAStatus) => (
          <Col xs={24} sm={12} lg={8} key={status.ota}>

}

function getStatusColor(status: string): string {
  switch (status) {
    case 'healthy': return 'green'
    case 'warning': return 'orange'
    case 'error': return 'red'
    default: return 'default'
  }
}

// app/api/ota/status/[hotelId]/route.ts
import { NextRequest } from 'next/server'
import { prisma } from '@/lib/prisma'

export async function GET(req: NextRequest, { params }: { params: { hotelId: string } }) {
  const hotelId = params.hotelId
  
  const metrics = await prisma.$queryRaw`
    SELECT 
      oc.ota,
      COALESCE(wm.last_sync, 'never') as last_sync,
      COALESCE(AVG(EXTRACT(EPOCH FROM (we.finished_at - we.started_at)) * 1000), 0)::int as latency_ms,
      COALESCE(SUM(CASE WHEN we.status = 'COMPLETED' THEN 1 ELSE 0 END), 0) as reservations_synced,
      COALESCE(SUM(CASE WHEN we.status = 'FAILED' THEN 1 ELSE 0 END), 0) as errors_24h,
      100 - COALESCE((rl.remaining_tokens::int), 0) as rate_limit_remaining
    FROM ota_config oc
    LEFT JOIN workflow_executions we ON we.tenant_id = oc.hotel_id AND we.workflow_id LIKE '%ota%'
    LEFT JOIN workflow_metrics wm ON wm.workflow_id LIKE '%ota%' AND wm.tenant_id = oc.hotel_id
    LEFT JOIN redis_rate_limit rl ON rl.key = concat(oc.hotel_id, '_', oc.ota)
    WHERE oc.hotel_id = ${hotelId}
      AND we.started_at > NOW() - INTERVAL '24 hours'
    GROUP BY oc.ota, wm.last_sync, rl.remaining_tokens
    ORDER BY reservations_synced DESC
  `

  return Response.json(metrics)
}

// services/voice-service.ts
import OpenAI from 'openai'
import { createClient } from '@deepgram/sdk'

export class VoiceService {
  private openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY })
  private deepgram = createClient(process.env.DEEPGRAM_API_KEY!)

  async processAudio(audioUrl: string): Promise<{ text: string; confidence: number }> {
    // 1. Transcrição (Deepgram)
    const transcription = await this.deepgram.transcription.preRecorded(
      { url: audioUrl },
      { model: 'nova-2', language: 'pt' }
    )

    const text = transcription.results.channels[0].alternatives[0].transcript
    
    // 2. Extrair intent (OpenAI GPT-4o-mini)
    const intent = await this.openai.chat.completions.create({
      model: 'gpt-4o-mini',
      messages: [{
        role: 'user',
        content: `
          EXTRAIA informações de RESERVA HOTEL do áudio:
          "${text}"
          
          Retorne JSON válido:
          {
            "guestName": "João Silva",
            "checkin": "2026-01-20",
            "checkout": "2026-01-22", 
            "roomType": "DELUXE",
            "adults": 2,
            "children": 0,
            "phone": "+5511999999999",
            "paymentMethod": "PIX"
          }
        `
      }]
    })

    const parsedIntent = JSON.parse(intent.choices[0].message.content!)
    
    return {
      text,
      confidence: transcription.results.channels[0].alternatives[0].confidence,
      intent: parsedIntent
    }
  }
}
