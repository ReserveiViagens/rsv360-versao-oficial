// api/financial-dashboard.js
import express from 'express'
import { PrismaClient } from '@prisma/client'

const app = express()
const prisma = new PrismaClient()

app.get('/api/finance/:tenantId/kpis', async (req, res) => {
  const { tenantId } = req.params
  
  const kpis = await prisma.$queryRaw`
    SELECT 
      COALESCE(SUM(total), 0)::numeric as revenue_30d,
      COALESCE(AVG(total / nights), 0)::numeric as adr,
      COUNT(*)::float / 45 * 100 as occupancy_pct, -- 45 quartos
      (SUM(total) / 45)::numeric as revpar,
      (SUM(total * 0.75)) / 45 as goppar, -- 75% margem
      COUNT(*) FILTER (WHERE status = 'CANCELLED')::float / COUNT(*) * 100 as cancellation_rate
    FROM reservations 
    WHERE hotel_id = ${tenantId}
      AND checkin >= NOW() - INTERVAL '30 days'
      AND status IN ('PAID', 'CHECKED_IN')
  `
  
  res.json(kpis[0])
})

app.listen(3001, () => console.log('💰 Financial API na porta 3001'))

// components/FinancialDashboard.tsx
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card'
import { Badge } from '@/components/ui/badge'
import { useQuery } from '@tanstack/react-query'
import { TrendingUp, TrendingDown, DollarSign } from 'lucide-react'

export default function FinancialDashboard({ tenantId }: { tenantId: string }) {
  const { data: kpis } = useQuery({
    queryKey: ['kpis', tenantId],
    queryFn: () => fetch(`/api/finance/${tenantId}/kpis`).then(res => res.json())
  })

  return (
    <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
      <KPI 
        title="RevPAR" 
        value={`R$ ${kpis?.revpar?.toFixed(0)}`}
        target={285}
        trend={kpis?.revpar > 285 ? 'up' : 'down'}
        color="bg-gradient-to-r from-emerald-400 to-green-500"
      />
      <KPI 
        title="ADR" 
        value={`R$ ${kpis?.adr?.toFixed(0)}`}
        target={380}
        trend="up"
        color="bg-gradient-to-r from-blue-400 to-indigo-500"
      />
      <KPI 
        title="Ocupação" 
        value={`${kpis?.occupancy_pct?.toFixed(1)}%`}
        target={78}
        trend="up"
        color="bg-gradient-to-r from-orange-400 to-yellow-500"
      />
    </div>
  )
}

function KPI({ title, value, target, trend, color }: KPIProps) {
  const isGood = parseFloat(value.replace(/[^\d.]/g, '')) >= target
  
  return (
    <Card className="hover:shadow-xl transition-all">
      <CardHeader>
        <CardTitle className="text-lg font-bold">{title}</CardTitle>
      </CardHeader>
      <CardContent>
        <div className={`text-3xl font-black p-4 rounded-xl mb-4 ${color}`}>
          {value}
        </div>
        <div className="flex items-center justify-between">
          <Badge variant={isGood ? 'default' : 'secondary'}>
            Target: {target}
          </Badge>
          {trend === 'up' ? <TrendingUp className="w-5 h-5 text-green-500" /> : <TrendingDown className="w-5 h-5 text-red-500" />}
        </div>
      </CardContent>
    </Card>
  )
}
