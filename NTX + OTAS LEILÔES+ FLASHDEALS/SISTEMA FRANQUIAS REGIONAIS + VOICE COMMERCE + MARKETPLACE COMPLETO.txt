        Leilões 70% OFF aumentaram 47% receita em Caldas Novas.
        Posso enviar demonstração WhatsApp agora?
      </Say>
      <Gather input="speech" action="/voice-response" timeout="10"/>
    </Response>`,
    to: phoneNumber,
    from: '+5511999999999'
  });

  // GPT-4o processa resposta
  client.calls(call.sid).fetch().then(async call => {
    const transcription = await getTranscription(call.sid);
    const response = await openai.chat.completions.create({
      model: 'gpt-4o',
      messages: [{ role: 'user', content: `Hotel disse: "${transcription}". Responda natural:` }]
    });
    
    // WhatsApp follow-up
    await sendWhatsApp(phoneNumber.replace('+55', ''), response.choices.message.content);
  });
}

// google-hotel-feed.js - 500 hotéis
async function generateGoogleFeed() {
  const hotels = await prisma.hotel.findMany({ 
    where: { active: true, marketplace: true }
  });

  const feed = hotels.map(hotel => ({
    HotelID: `rsv360_${hotel.id}`,
    Name: hotel.name,
    City: hotel.city,
    RatePlans: hotel.ratePlans.map(rp => ({
      Price: rp.leilaoPrice || rp.standardPrice,
      Availability: rp.dates,
      DirectBookingLink: `https://marketplace.rsv360.com/hotel/${hotel.id}`
    }))
  }));

  // Upload Google Hotel Center
  await fetch('https://hotelcenter.googleapis.com/v1/hotels:batchUpdate', {
    method: 'POST',
    headers: { Authorization: `Bearer ${GOOGLE_TOKEN}` },
    body: JSON.stringify({ parent: 'properties/rsv360-br-001', hotels: feed })
  });
}
