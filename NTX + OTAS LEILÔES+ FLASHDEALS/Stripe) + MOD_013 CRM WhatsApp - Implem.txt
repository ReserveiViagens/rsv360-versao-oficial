// services/payment-service.ts
import Stripe from 'stripe'

export class PaymentService {
  private stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
    apiVersion: '2024-10-01.acacia'
  })

  async createPaymentIntent(reservationId: string, amount: number, customerPhone: string) {
    // 1. Validar reserva
    const reservation = await prisma.reservation.findUnique({
      where: { id: reservationId }
    })
    
    if (!reservation || reservation.status !== 'PENDING_PAYMENT') {
      throw new Error('Reserva inválida para pagamento')
    }

    // 2. Criar PaymentIntent (Pix + Cartão)
    const paymentIntent = await this.stripe.paymentIntents.create({
      amount: Math.round(amount * 100), // BRL cents
      currency: 'brl',
      payment_method_types: ['card', 'pix'],
      metadata: {
        reservation_id: reservationId,
        hotel_id: reservation.hotelId,
        customer_phone: customerPhone
      },
      description: `Reserva RSV360 #${reservationId}`
    })

    // 3. Gerar QR Pix se necessário
    if (paymentIntent.next_action?.type === 'use_stripe_sdk') {
      const pixData = await this.stripe.paymentIntents.retrieve(paymentIntent.id)
      return {
        clientSecret: paymentIntent.client_secret!,
        pixKey: pixData.next_action.pix_display_details.pix_code,
        pixImage: pixData.next_action.pix_display_details.pix_qr_code,
        expiresAt: pixData.next_action.pix_display_details.expires_at
      }
    }

    return { clientSecret: paymentIntent.client_secret! }
  }

  async handleStripeWebhook(payload: any, signature: string) {
    const event = this.stripe.webhooks.constructEvent(
      payload, 
      signature, 
      process.env.STRIPE_WEBHOOK_SECRET!
    )

    if (event.type === 'payment_intent.succeeded') {
      const paymentIntent = event.data.object as Stripe.PaymentIntent
      const reservationId = paymentIntent.metadata.reservation_id
      
      // Atualizar reserva como paga
      await prisma.reservation.update({
        where: { id: reservationId },
        data: { 
          status: 'PAID',
          stripePaymentId: paymentIntent.id,
          paidAt: new Date()
        }
      })

      // Trigger workflow "send_receipt"
      await fetch('http://workflow-engine:3001/workflows/payment_complete/execute', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          tenantId: paymentIntent.metadata.hotel_id,
          input: { reservationId, paymentIntentId: paymentIntent.id }
        })
      })
    }
  }
}

// services/whatsapp-crm.ts
import { CustomerInteraction, PrismaClient } from '@prisma/client'

export class WhatsAppCRM {
  constructor(private prisma: PrismaClient) {}

  async handleIncomingMessage(whatsappMsg: any) {
    const { from, body, messageId } = whatsappMsg
    
    // 1. Buscar/criar customer
    let customer = await this.prisma.customer.upsert({
      where: { phone: from },
      update: {},
      create: { 
        tenantId: 'hotel_123', 
        phone: from 
      }
    })

    // 2. Classificar intent (GPT-4o-mini)
    const intent = await this.classifyMessageIntent(body)
    
    // 3. Criar interaction
    const interaction = await this.prisma.customerInteraction.create({
      data: {
        customerId: customer.id,
        type: intent.type,
        channel: 'WHATSAPP',
        direction: 'INCOMING',
        content: { text: body, whatsappId: messageId },
        status: 'DELIVERED'
      }
    })

    // 4. Responder automaticamente
    if (intent.type === 'RESERVATION_REQUEST') {
      await this.sendReservationOptions(customer.phone, intent.data)
    }

    return interaction
  }

  private async classifyMessageIntent(message: string) {
    const response = await openai.chat.completions.create({
      model: 'gpt-4o-mini',
      messages: [{
        role: 'user',
        content: `
          Classifique esta mensagem WhatsApp como intent de hotel:
          "${message}"
          
          Retorne JSON:
          {
            "type": "RESERVATION_REQUEST|CHECK_AVAILABILITY|PRICE_INQUIRY|CONTACT|SUPPORT",
            "data": { checkin, checkout, adults, roomType }
          }
        `
      }]
    })

    return JSON.parse(response.choices[0].message.content!)
  }

  async sendReservationConfirmation(phone: string, reservation: any) {
    const message = `
✅ *RESERVA CONFIRMADA* RSV360

👤 ${reservation.guestName}
📅 ${new Date(reservation.checkin).toLocaleDateString('pt-BR')} → ${new Date(reservation.checkout).toLocaleDateString('pt-BR')}
🛏️ ${reservation.roomType}
💰 R$ ${reservation.total.toLocaleString('pt-BR')}

*Pix copia e cola:* ${reservation.pixCopyPaste}
*Vencimento:* ${new Date(reservation.expiresAt).toLocaleString('pt-BR')}

👉 https://dashboard.rsv360.com/reserva/${reservation.id}
    `

    return this.sendWhatsApp(phone, message)
  }
}

// services/auction-service.ts
export class AuctionService {
  async placeBid(auctionId: string, customerPhone: string, amount: number) {
    const auction = await prisma.auction.findUnique({
      where: { id: auctionId }
    })

    if (!auction || auction.status !== 'ACTIVE') {
      throw new Error('Leilão não ativo')
    }

    if (amount < auction.reservePrice) {
      throw new Error('Lance abaixo do preço mínimo')
    }

    // Idempotência - último lance do mesmo user
    const lastBid = await prisma.auctionBid.findFirst({
      where: { auctionId, customerId: customerPhone },
      orderBy: { createdAt: 'desc' }
    })

    if (lastBid && lastBid.amount >= amount) {
      throw new Error('Lance deve ser maior que o anterior')
    }

    // Criar bid
    const bid = await prisma.auctionBid.create({
      data: {
        auctionId,
        customerId: customerPhone,
        amount
      }
    })

    // Atualizar maior lance
    await prisma.auction.update({
      where: { id: auctionId },
      data: { currentBid: amount }
    })

    // Notificar via WhatsApp TODOS os bidders
    await this.notifyBidders(auctionId, customerPhone, amount)

    return bid
  }

  private async notifyBidders(auctionId: string, winnerPhone: string, winningAmount: number) {
    const recentBidders = await prisma.auctionBid.findMany({
      where: { auctionId },
      orderBy: { createdAt: 'desc' },
      take: 10
    })

    for (const bidder of recentBidders) {
      const message = bidder.customerId === winnerPhone 
        ? `🎉 VOCÊ É O VENCEDOR! Leilão ${auctionId}. Pague R$ ${winningAmount} agora!`
        : `⚡ Novo lance de R$ ${winningAmount}! Continue licitando!`

      await whatsappCRM.sendMessage(bidder.customerId, message)
    }
  }
}

// app/api/finance/[tenantId]/route.ts
export async function GET(req: Request, { params }: { params: { tenantId: string } }) {
  const tenantId = params.tenantId
  
  const metrics = await prisma.$queryRaw`
    SELECT 
      SUM(r.total) as revenue,
      SUM(r.total - COALESCE(r.otaCommission, 0)) as net_revenue,
      COUNT(r.id) as reservations,
      AVG(r.total / NULLIF(r.nights, 0)) as adr,
      (SUM(r.total) / 150.0)::numeric as revpar, -- 150 quartos
      COUNT(r.id) FILTER (WHERE r.status = 'CANCELLED') as cancellations
    FROM reservations r
    WHERE r.hotel_id = ${tenantId} 
      AND r.created_at >= NOW() - INTERVAL '30 days'
  `

  return Response.json(metrics[0])
}
