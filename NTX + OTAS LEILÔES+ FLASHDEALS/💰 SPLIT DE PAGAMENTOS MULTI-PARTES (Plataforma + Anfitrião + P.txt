CREATE TABLE booking_splits (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  booking_id UUID REFERENCES bookings(id) ON DELETE CASCADE,
  platform_amount DECIMAL(10,2),
  host_amount DECIMAL(10,2),
  partner_amount DECIMAL(10,2) DEFAULT 0,
  partner_id UUID REFERENCES partners(id),
  platform_fee_percent DECIMAL(5,2) DEFAULT 20.00,
  host_fee_percent DECIMAL(5,2) DEFAULT 75.00,
  partner_fee_percent DECIMAL(5,2) DEFAULT 5.00,
  total_amount DECIMAL(10,2) NOT NULL,
  status TEXT DEFAULT 'pending', -- pending, split_completed, disputed
  created_at TIMESTAMP DEFAULT NOW()
);

-- Transações de split (histórico)
CREATE TABLE split_transactions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  booking_split_id UUID REFERENCES booking_splits(id),
  recipient_type TEXT, -- 'platform', 'host', 'partner'
  recipient_id UUID,
  amount DECIMAL(10,2),
  status TEXT, -- paid, failed, refunded
  gateway_transaction_id TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);
💳 2. LÓGICA DE SPLIT INTELIGENTE /lib/paymentSplits.ts
typescript
import { supabaseAdmin } from './supabase'

export interface SplitConfig {
  totalAmount: number
  hostId: string
  partnerId?: string
  platformFeePercent: number // 20%
  hostFeePercent: number     // 75%
  partnerFeePercent?: number // 5%
}

export async function calculateAndCreateSplit(config: SplitConfig) {
  // Calcular splits
  const platformAmount = config.totalAmount * (config.platformFeePercent / 100)
  const hostAmount = config.totalAmount * (config.hostFeePercent / 100)
  const partnerAmount = config.partnerId 
    ? config.totalAmount * (config.partnerFeePercent! / 100)
    : 0

  // Criar registro de split
  const { data: split } = await supabaseAdmin
    .from('booking_splits')
    .insert({
      booking_id: config.bookingId, // passado externamente
      platform_amount: platformAmount,
      host_amount: hostAmount,
      partner_amount: partnerAmount,
      partner_id: config.partnerId || null,
      platform_fee_percent: config.platformFeePercent,
      host_fee_percent: config.hostFeePercent,
      partner_fee_percent: config.partnerFeePercent,
      total_amount: config.totalAmount
    })
    .select()
    .single()

  return {
    splitId: split.id,
    splits: {
      platform: platformAmount,
      host: hostAmount,
      partner: partnerAmount
    }
  }
}

export async function executeSplit(splitId: string, gateway: 'stripe' | 'mercadopago') {
  const { data: split } = await supabaseAdmin
    .from('booking_splits')
    .select('*')
    .eq('id', splitId)
    .single()

  if (!split) throw new Error('Split not found')

  // Stripe Connect Transfers
  if (gateway === 'stripe') {
    // Plataforma recebe total → transfiere para host/partner
    await stripe.transfers.create({
      amount: Math.round(split.host_amount * 100),
      currency: 'brl',
      destination: 'acct_host_stripe_id', // ID Stripe Connect do host
      transfer_group: splitId
    })

    if (split.partner_amount > 0) {
      await stripe.transfers.create({
        amount: Math.round(split.partner_amount * 100),
        currency: 'brl',
        destination: 'acct_partner_stripe_id',
        transfer_group: splitId
      })
    }
  }

  // Mercado Pago Splits
  if (gateway === 'mercadopago') {
    // Mercado Pago Pro + splits
    await mercadopago.split.create({
      marketplace_fee: split.platform_amount,
      primary_receiver: { id: 'host_user_id', percentage: 75 },
      secondary_receivers: split.partner_id ? [{
        id: 'partner_user_id',
        percentage: 5
      }] : []
    })
  }

  // Marcar como concluído
  await supabaseAdmin
    .from('booking_splits')
    .update({ status: 'split_completed' })
    .eq('id', splitId)
}
🖥️ 3. API ROUTE COM SPLIT /app/api/bookings/create-with-split/route.ts
typescript
import { NextRequest, NextResponse } from 'next/server'
import { createBooking } from '@/lib/bookings'
import { calculateAndCreateSplit } from '@/lib/paymentSplits'

export async function POST(request: NextRequest) {
  try {
    const data = await request.json()
    
    const { splits, splitId } = await calculateAndCreateSplit({
      totalAmount: data.total_price,
      hostId: data.host_id,
      partnerId: data.partner_id,
      platformFeePercent: 20,
      hostFeePercent: 75,
      partnerFeePercent: data.partner_id ? 5 : 0,
      bookingId: data.booking_id // gerado antes
    })

    // Criar booking com split
    const bookingResult = await createBooking({
      ...data,
      split_id: splitId
    }, data.gateway)

    return NextResponse.json({
      success: true,
      booking: bookingResult.booking,
      splits,
      paymentUrl: bookingResult.payment.url
    })
  } catch (error) {
    return NextResponse.json({ error: 'Split creation failed' }, { status: 500 })
  }
}
