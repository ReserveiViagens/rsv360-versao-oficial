// api/marketplace/search.js
export default async function handler(req, res) {
  const { q, date } = req.query;
  
  const hotels = await prisma.hotel.findMany({
    where: {
      city: { contains: q },
      availability: { has: date }
    },
    orderBy: [
      { revpar: 'desc' },      // 50%
      { ocupacao: 'desc' },    // 30%
      { reviews: 'desc' }      // 20%
    ],
    take: 24
  });

  // Calcular comissão 8%
  const withCommission = hotels.map(h => ({
    ...h,
    marketplaceFee: h.price * 0.08,  // R$17,60 de R$220
    hotelRecebe: h.price * 0.92      // R$202,40
  }));

  res.json(withCommission);
}

// daily-payout.js
async function dailyHotelPayout() {
  const yesterdayReservations = await prisma.reservation.findMany({
    where: { createdAt: { gte: yesterday } }
  });
  
  for (const res of yesterdayReservations) {
    const fee = res.total * 0.08;
    const hotelShare = res.total * 0.92;
    
    // Repasse hotel
    await stripe.transfers.create({
      amount: hotelShare * 100,
      currency: 'brl',
      destination: res.hotel.stripeAccount,
      metadata: { reservationId: res.id }
    });
    
    // Afiliado (20% fee)
    if (res.affiliateId) {
      await stripe.transfers.create({
        amount: fee * 0.20 * 100,
        destination: res.affiliate.stripeAccount
      });
    }
  }
}
