🗄️ DATABASE - NOVAS TABELAS
Tabela: propriedades_caldas_novas
sql
CREATE TABLE propriedades_caldas_novas (
  id SERIAL PRIMARY KEY,
  nome_oficial VARCHAR(255) NOT NULL,
  tipo VARCHAR(50),  -- hotel, resort, flat, apartamento, casa, loft, chalé, casarão
  categoria VARCHAR(20),  -- básico, standard, superior, deluxe, premium
  endereco TEXT NOT NULL,
  latitude DECIMAL(10,8),
  longitude DECIMAL(11,8),
  
  -- Informações de Contato
  telefone VARCHAR(20),
  email VARCHAR(255),
  website VARCHAR(255),
  gerente_nome VARCHAR(255),
  gerente_telefone VARCHAR(20),
  
  -- Capacidade
  total_quartos INTEGER,
  capacidade_maxima INTEGER,
  tipos_quarto JSONB,  -- {
                        --   "standard": 10,
                        --   "deluxe": 5,
                        --   "suíte": 3
                        -- }
  
  -- Avaliação
  avaliacao_media DECIMAL(3,1),
  total_avaliacoes INTEGER,
  
  -- Preços
  preco_minimo DECIMAL(10,2),
  preco_maximo DECIMAL(10,2),
  preco_medio DECIMAL(10,2),
  
  -- Comodidades
  comodidades TEXT[],  -- piscina_termal, piscina_fria, sauna, academia, restaurante, bar, estacionamento
  
  -- Imagens
  imagens_url TEXT[],
  logo_url VARCHAR(255),
  
  -- Status
  ativo BOOLEAN DEFAULT true,
  data_cadastro TIMESTAMP DEFAULT NOW(),
  data_atualizacao TIMESTAMP DEFAULT NOW(),
  
  -- Integração
  pms_integrado BOOLEAN DEFAULT false,
  pms_tipo VARCHAR(50),  -- opera, cloudbeds, hostaway
  pms_api_key VARCHAR(255) ENCRYPTED,
  
  -- Leilão
  leilao_ativo BOOLEAN DEFAULT true,
  lance_minimo DECIMAL(10,2),
  
  UNIQUE(nome_oficial, endereco)
);
Tabela: quartos_caldas_novas
sql
CREATE TABLE quartos_caldas_novas (
  id SERIAL PRIMARY KEY,
  propriedade_id INTEGER REFERENCES propriedades_caldas_novas(id),
  numero_quarto VARCHAR(10),
  tipo VARCHAR(50),  -- standard, deluxe, suíte
  capacidade INTEGER,
  
  -- Amenidades do quarto
  amenidades TEXT[],  -- ar_condicionado, tv, wifi, frigobar, cozinha
  
  -- Disponibilidade
  ativo BOOLEAN DEFAULT true,
  data_bloqueio_inicio DATE,
  data_bloqueio_fim DATE,
  
  -- Preço
  preco_noite DECIMAL(10,2),
  taxa_servico DECIMAL(10,2),
  
  created_at TIMESTAMP DEFAULT NOW()
);
Tabela: leiloes_caldas_novas (Nova)
sql
CREATE TABLE leiloes_caldas_novas (
  id SERIAL PRIMARY KEY,
  propriedade_id INTEGER REFERENCES propriedades_caldas_novas(id),
  tipo_leilao VARCHAR(50),  -- quarto_individual, bloco_quartos, casa_inteira, pacote
  
  -- Detalhes do Leilão
  titulo VARCHAR(255),
  descricao TEXT,
  data_inicio TIMESTAMP,
  data_fim TIMESTAMP,
  data_retirada DATE,
  data_devolucao DATE,
  
  -- Lances
  lance_minimo DECIMAL(10,2),
  lance_atual DECIMAL(10,2),
  numero_lances INTEGER DEFAULT 0,
  
  -- Configuração
  quantidade_quartos INTEGER,  -- se bloco
  taxa_servico DECIMAL(10,2),
  cancelamento_permitido BOOLEAN DEFAULT true,
  
  status VARCHAR(50),  -- ativo, pausado, encerrado, vendido
  vencedor_id INTEGER,
  
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);
