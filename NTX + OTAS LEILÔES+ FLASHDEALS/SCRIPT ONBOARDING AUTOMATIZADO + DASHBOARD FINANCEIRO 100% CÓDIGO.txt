API Route - KPIs Consolidados
typescript
// app/api/finance/[tenantId]/kpis/route.ts
import { NextRequest } from 'next/server'
import { prisma } from '@/lib/prisma'

export async function GET(req: NextRequest, { params }: { params: { tenantId: string } }) {
  const tenantId = params.tenantId
  
  const kpis = await prisma.$queryRaw`
    WITH daily_stats AS (
      SELECT 
        DATE(checkin) as date,
        COUNT(*) as reservations,
        SUM(total) as revenue,
        SUM(total - COALESCE(ota_commission, 0)) as net_revenue,
        COUNT(*)::float / 45.0 as occupancy_pct, -- 45 quartos padrão Caldas
        AVG(total::numeric / nights)::numeric as adr,
        SUM(total) / 45.0 as revpar,
        SUM(total - COALESCE(ota_commission, 0) - cleaning_cost) / 45.0 as goppar
      FROM reservations 
      WHERE hotel_id = ${tenantId}
        AND status IN ('PAID', 'CHECKED_IN')
        AND checkin >= NOW() - INTERVAL '30 days'
      GROUP BY DATE(checkin)
    )
    SELECT 
      SUM(revenue)::numeric as total_revenue_30d,
      SUM(net_revenue)::numeric as total_net_30d,
      AVG(adr) as avg_adr,
      AVG(occupancy_pct * 100)::numeric as avg_occupancy_pct,
      AVG(revpar)::numeric as avg_revpar,
      AVG(goppar)::numeric as avg_goppar,
      (SUM(revenue) / 45.0 / 30)::numeric as trevpar,
      COUNT(*) FILTER (WHERE status = 'CANCELLED')::float / COUNT(*) * 100 as cancellation_rate_pct
    FROM daily_stats
  `

  return Response.json(kpis[0])
}
Dashboard Financeiro Completo (React)
tsx
// components/dashboard/FinancialDashboard.tsx
'use client'

import { Card, Statistic, Progress, Row, Col, Table, Badge, Button } from 'antd'
import { TrendingUp, TrendingDown, DollarSign, Bed, Hotel } from 'lucide-react'
import { useQuery } from '@tanstack/react-query'
import { useEffect, useState } from 'react'

interface FinancialKPI {
  total_revenue_30d: number
  avg_revpar: number
  avg_adr: number
  avg_occupancy_pct: number
  avg_goppar: number
  trevpar: number
  cancellation_rate_pct: number
}

export default function FinancialDashboard({ tenantId }: { tenantId: string }) {
  const [range, setRange] = useState<'today' | '7d' | '30d'>('30d')
  
  const { data: kpis, refetch } = useQuery<FinancialKPI>({
    queryKey: ['finance-kpis', tenantId, range],
    queryFn: () => 
      fetch(`/api/finance/${tenantId}/kpis?range=${range}`).then(res => res.json()),
    refetchInterval: 30000 // 30s real-time
  })

  const kpiCards = [
    {
      title: 'RevPAR',
      value: `R$ ${kpis?.avg_revpar?.toFixed(0) || 0}`,
      target: 285,
      color: kpis?.avg_revpar! > 285 ? 'success' : 'warning'
    },
    {
      title: 'ADR', 
      value: `R$ ${kpis?.avg_adr?.toFixed(0) || 0}`,
      target: 380,
      color: kpis?.avg_adr! > 380 ? 'success' : 'warning'
    },
    {
      title: 'Ocupação',
      value: `${kpis?.avg_occupancy_pct?.toFixed(1) || 0}%`,
      target: 78,
      color: kpis?.avg_occupancy_pct! > 78 ? 'success' : 'warning'
    }
  ]

  return (
    <div className="space-y-8">
      {/* Header + Range Selector */}
      <div className="flex justify-between items-center">
        <h2 className="text-3xl font-bold text-gray-900">Dashboard Financeiro</h2>
        <div className="flex space-x-2">
          {['today', '7d', '30d'].map(period => (
            <Button 
              key={period}
              type={range === period ? 'primary' : 'default'}
              onClick={() => setRange(period as any)}
            >
              {period === 'today' ? 'Hoje' : period === '7d' ? '7 dias' : '30 dias'}
            </Button>
          ))}
        </div>
      </div>

      {/* KPIs Principais */}
      <Row gutter={[24, 24]}>
        {kpiCards.map((kpi, i) => (
          <Col span={8} key={i}>
            <Card>
              <Statistic
                title={kpi.title}
                value={kpi.value}
                prefix={<Hotel className="w-5 h-5" />}
                valueStyle={{ color: kpi.color === 'success' ? '#52c41a' : '#faad14' }}
              />
              <div className="mt-2 flex items-center text-sm">
                {kpi.target && (
                  <>
                    <span>Target: {typeof kpi.value === 'string' ? kpi.value.replace('R$ ', 'R$ ') : `R$ ${kpi.target}`}</span>
                    <div className="ml-auto">
                      {Number(kpi.value.replace('%', '').replace('R$ ', '')) > kpi.target ? 
                        <TrendingUp className="w-4 h-4 text-green-500" /> : 
                        <TrendingDown className="w-4 h-4 text-red-500" />
                      }
                    </div>
                  </>
                )}
              </div>
            </Card>
          </Col>
        ))}
      </Row>

      {/* Receita 30d + Cancelamentos */}
      <Row gutter={24}>
        <Col span={12}>
          <Card title="Receita 30 Dias">
            <Statistic
              value={kpis?.total_revenue_30d?.toLocaleString('pt-BR', { style: 'currency', currency: 'BRL' })}
              prefix={<DollarSign className="w-6 h-6" />}
              valueStyle={{ color: '#1677ff' }}
            />
          </Card>
        </Col>
        <Col span={12}>
          <Card title="Cancelamentos">
            <Statistic
              value={`${kpis?.cancellation_rate_pct?.toFixed(1) || 0}%`}
              suffix={<span className="text-red-500">Target <15%</span>}
              valueStyle={{ color: kpis?.cancellation_rate_pct! > 15 ? '#ff4d4f' : '#52c41a' }}
            />
          </Card>
        </Col>
      </Row>
    </div>
  )
}
