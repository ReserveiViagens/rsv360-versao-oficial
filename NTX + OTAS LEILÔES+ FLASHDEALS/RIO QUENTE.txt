// scoreHotel.js - Priorização automática
function scoreHotel(hotel) {
  let score = 0;
  
  // Quartos (40%)
  score += Math.min(hotel.rooms / 45 * 40, 40);
  
  // ADR (30%)
  score += Math.min((hotel.adr / 380) * 30, 30);
  
  // Reviews (20%)
  score += Math.min((hotel.reviews / 8) * 20, 20);
  
  // WhatsApp (10%)
  score += hotel.hasWhatsapp ? 10 : 0;
  
  return {
    score: Math.round(score),
    priority: score >= 80 ? 'TIER1' : score >= 60 ? 'TIER2' : 'TIER3',
    mrrPotential: Math.round((hotel.rooms * hotel.adr * 0.85) / 30)
  };
}
