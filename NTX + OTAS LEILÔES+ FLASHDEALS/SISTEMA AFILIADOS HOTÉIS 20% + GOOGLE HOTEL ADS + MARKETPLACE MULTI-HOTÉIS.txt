javascript
// afiliados-rsv360.js - Production Ready
const express = require('express');
const stripe = require('stripe')(process.env.STRIPE_SECRET);
const prisma = new PrismaClient();

class AfiliadosSystem {
  
  // Dashboard afiliado
  app.get('/af/:id/dashboard', async (req, res) => {
    const affiliate = await prisma.affiliate.findUnique({
      where: { id: req.params.id },
      include: { hotels: true }
    });
    
    res.json({
      totalMRR: affiliate.hotels.reduce((sum, h) => sum + h.mrr, 0),
      commission: affiliate.totalMRR * 0.20,
      hotelsCount: affiliate.hotels.length,
      link: `rsv360.com/af/${req.params.id}`
    });
  });

  // Payout mensal automático
  async payoutMonth() {
    const affiliates = await prisma.affiliate.findMany({
      where: { hotels: { some: {} } }
    });
    
    for (const affiliate of affiliates) {
      const commission = affiliate.totalMRR * 0.20 * 100; // cents
      
      await stripe.transfers.create({
        amount: commission,
        currency: 'brl',
        destination: affiliate.stripeAccountId,
        description: `Comissão RSV360 ${new Date().getMonth() + 1}/2026`
      });
      
      console.log(`✅ ${affiliate.name}: R$${commission/100} pago`);
    }
  }
}
