3. Implementação Retries + Backoff Exponencial
typescript
// engine/retry.ts
export async function executeWithRetry(
  step: WorkflowStepExecution,
  executor: () => Promise<any>,
  config: HttpStepConfig
): Promise<any> {
  const maxRetries = config.retry?.maxRetries ?? 3
  const baseDelay = config.retry?.baseDelayMs ?? 1000
  const isExponential = config.retry?.backoff !== 'linear'
  
  for (let attempt = 0; attempt <= maxRetries; attempt++) {
    try {
      step.status = 'RUNNING'
      step.retryCount = attempt
      await prisma.workflowStepExecution.update({ where: { id: step.id }, data: step })
      
      const result = await executor()
      
      step.status = 'COMPLETED'
      step.output = result
      await prisma.workflowStepExecution.update({ where: { id: step.id }, data: step })
      return result
      
    } catch (error) {
      step.error = error.message
      await prisma.workflowStepExecution.update({ where: { id: step.id }, data: step })
      
      if (attempt === maxRetries) {
        step.status = 'FAILED'
        throw error
      }
      
      // Backoff exponencial: 1s, 2s, 4s, 8s...
      const delay = isExponential 
        ? baseDelay * Math.pow(2, attempt)
        : baseDelay * (attempt + 1)
        
      await new Promise(r => setTimeout(r, delay))
    }
  }
}

export async function createCheckpoint(
  executionId: string,
  stepIndex: number,
  currentState: any
): Promise<void> {
  await prisma.workflowCheckpoint.create({
    data: {
      executionId,
      stepIndex,
      data: currentState
    }
  })

export async function resumeFromCheckpoint(
  executionId: string,
  stepIndex: number
): Promise<any> {
  const checkpoint = await prisma.workflowCheckpoint.findUnique({
    where: { executionId_stepIndex: { executionId, stepIndex } }
  })
  return checkpoint?.data ?? {}
}

// Salva estado após CADA step crítico
export async function saveExecutionProgress(execution: WorkflowExecution) {
  await createCheckpoint(
    execution.id, 
    execution.currentStep, 
    execution.output
  )
  
  await prisma.workflowExecution.update({
    where: { id: execution.id },
    data: {
      currentStep: execution.currentStep,
      output: execution.output
    }
  })
}
5. Orquestração com Webhooks/Callbacks
typescript
// engine/orchestrator.ts
interface WebhookStepConfig {
  type: 'webhook'
  stepId: string
  callbackUrl: string  // "https://seu-sistema.com/webhooks/continue"
  payload: JsonTemplate
  timeoutMs: number    // Quando aguardar callback
  continueOnTimeout?: boolean
}

export async function executeWebhookStep(
  execution: WorkflowExecution,
  step: WorkflowStepExecution,
  config: WebhookStepConfig
): Promise<void> {
  // 1. Persistir estado atual (checkpoint)
  await saveExecutionProgress(execution)
  
  // 2. Enviar webhook para sistema externo
  const webhookPayload = templateRender(config.payload, execution.output)
  await fetch(config.callbackUrl, {
    method: 'POST',
    headers: { 'X-Workflow-ID': execution.id },
    body: JSON.stringify({
      ...webhookPayload,
      workflowExecutionId: execution.id,
      nextStep: execution.currentStep + 1,
      callbackRequired: true
    })
  })
  
  // 3. Aguardar callback ou timeout
  step.status = 'WAITING_CALLBACK'
  await prisma.workflowStepExecution.update({ where: { id: step.id }, data: step })
  
  // 4. Scheduler monitora conclusão (BullMQ job)
  await scheduleCallbackMonitor(execution.id, config.timeoutMs)
}

export async function handleCallback(webhook: any) {
  const { workflowExecutionId, stepIndex, success, output } = webhook
  
  const execution = await prisma.workflowExecution.findUnique({
    where: { id: workflowExecutionId },
    include: { steps: true }
  })
  
  if (!execution) return
  
  // Atualizar step específico
  const step = execution.steps.find(s => s.stepIndex === stepIndex)
  if (step) {
    step.status = success ? 'COMPLETED' : 'FAILED'
    step.output = success ? output : null
    await prisma.workflowStepExecution.update({
      where: { id: step.id },
      data: step
    })
    
    // Continuar próximo step
    await continueExecution(workflowExecutionId)
  }
}

Algoritmo Backoff (Node.js):

javascript
async function scheduleRetry(step) {
  const attempt = step.retry_count + 1;
  const maxRetries = step.config.retry.maxRetries || 3;
  
  if (attempt > maxRetries) {
    step.status = 'FAILED';
    return;
  }
  
  // Backoff exponencial: 1s → 2s → 4s → 8s
  const delayMs = step.config.retry.baseDelayMs * (2 ** (attempt - 1));
  step.next_attempt_at = new Date(Date.now() + delayMs);
  step.retry_count = attempt;
  
  await db.workflow_steps.update(step.id, step);
}
Dead Letter Queue (Automático):

sql
-- Após 3 falhas, mover para DLQ
INSERT INTO workflow_dlq (execution_id, step_id, error, final_state)
SELECT execution_id, id, error, state 
FROM workflow_steps s
JOIN workflow_executions e ON s.execution_id = e.id
WHERE s.status = 'FAILED' AND s.retry_count >= s.max_retries;
🎯 Queries Operacionais Essenciais
sql
-- 1. Dashboard Execuções (últimas 24h)
SELECT w.name, COUNT(*) as total, 
       SUM(CASE WHEN status='COMPLETED' THEN 1 ELSE 0 END) as success,
       AVG(EXTRACT(EPOCH FROM (finished_at - started_at))) as avg_duration
FROM workflow_executions e
JOIN workflows w ON e.workflow_id = w.id
WHERE started_at > NOW() - INTERVAL '24 hours'
GROUP BY w.name;

-- 2. Workflows em falha (precisa retry)
SELECT * FROM workflow_steps 
WHERE status IN ('FAILED', 'RETRYING') 
  AND next_attempt_at <= NOW()
ORDER BY next_attempt_at ASC;

-- 3. Estado de uma execução específica
SELECT * FROM workflow_executions e
LEFT JOIN workflow_steps s ON e.id = s.execution_id
LEFT JOIN workflow_state st ON e.id = st.execution_id
WHERE e.id = 'execution_123';

// worker/index.ts - Worker que processa steps pendentes
import { Pool } from 'pg'
import pino from 'pino'
import { v4 as uuidv4 } from 'uuid'

const logger = pino({ level: 'info' })
const db = new Pool({ connectionString: process.env.DATABASE_URL })

interface WorkflowStep {
  id: string
  execution_id: string
  step_index: number
  step_id: string
  type: string
  config: any
  status: string
  retry_count: number
  max_retries: number
  next_attempt_at: string | null
}

async function processPendingSteps() {
  while (true) {
    try {
      // Buscar steps prontos para execução
      const steps: WorkflowStep[] = await db.query(`
        SELECT * FROM workflow_steps 
        WHERE status = 'PENDING' 
           OR (status = 'FAILED' AND retry_count < max_retries AND next_attempt_at <= NOW())
        ORDER BY next_attempt_at NULLS FIRST, created_at
        LIMIT 10 FOR UPDATE SKIP LOCKED
      `)

      for (const step of steps.rows) {
        await executeStep(step)
      }

      // Pausa 100ms entre batches
      await new Promise(r => setTimeout(r, 100))
      
    } catch (error) {
      logger.error(error, 'Erro no worker loop')
      await new Promise(r => setTimeout(r, 5000)) // Backoff 5s
    }
  }
}

async function executeStep(step: WorkflowStep) {
  try {
    logger.info({ stepId: step.id }, `Executando step ${step.step_id}`)
    
    // Marcar como RUNNING
    await db.query(
      `UPDATE workflow_steps SET status = 'RUNNING', started_at = NOW() WHERE id = $1`,
      [step.id]
    )

    let output: any

    switch (step.type) {
      case 'http':
        output = await executeHttpStep(step.config)
        break
      case 'delay':
        output = await executeDelayStep(step.config)
        break
      case 'internal':
        output = await executeInternalStep(step.config)
        break
      default:
        throw new Error(`Step type ${step.type} não implementado`)
    }

    // Sucesso
    await db.query(`
      UPDATE workflow_steps 
      SET status = 'COMPLETED', output = $1, finished_at = NOW() 
      WHERE id = $2
    `, [JSON.stringify(output), step.id])

    logger.info({ stepId: step.id }, 'Step concluído com sucesso')
    await continueNextStep(step.execution_id, step.step_index + 1)

  } catch (error: any) {
    await handleStepFailure(step, error)
  }
}

async function executeHttpStep(config: any): Promise<any> {
  const controller = new AbortController()
  const timeoutId = setTimeout(() => controller.abort(), config.timeoutMs || 10000)

  const response = await fetch(config.url, {
    method: config.method || 'GET',
    headers: config.headers || {},
    body: config.body ? JSON.stringify(config.body) : undefined,
    signal: controller.signal
  })

  clearTimeout(timeoutId)

  if (!response.ok) {
    throw new Error(`HTTP ${response.status}: ${response.statusText}`)
  }

  return response.json()
}

async function continueNextStep(executionId: string, nextStepIndex: number) {
  // Atualizar current_step na execução
  await db.query(`
    UPDATE workflow_executions 
    SET current_step = $1 
    WHERE id = $2 AND status = 'RUNNING'
  `, [nextStepIndex, executionId])

  // Se ainda tem steps pendentes, processar próximo
  const nextStep = await db.query(`
    SELECT * FROM workflow_steps 
    WHERE execution_id = $1 AND step_index = $2
  `, [executionId, nextStepIndex])

  if (nextStep.rows.length > 0) {
    // Próximo step será pego pelo worker
    logger.info({ executionId, nextStepIndex }, 'Próximo step agendado')
  } else {
    // Workflow concluído
    await db.query(`
      UPDATE workflow_executions 
      SET status = 'COMPLETED', finished_at = NOW() 
      WHERE id = $1
    `, [executionId])
  }
}

async function handleStepFailure(step: WorkflowStep, error: Error) {
  const retryCount = step.retry_count + 1
  const maxRetries = step.max_retries || 3
  
  if (retryCount >= maxRetries) {
    // Falha final
    await db.query(`
      UPDATE workflow_steps 
      SET status = 'FAILED', 
          error = $1, 
          finished_at = NOW(),
          retry_count = $2
      WHERE id = $3
    `, [error.message, retryCount, step.id])
    
    // Falhar execução inteira
    await db.query(`
      UPDATE workflow_executions 
      SET status = 'FAILED', finished_at = NOW()
      WHERE id = $1
    `, [step.execution_id])
    
  } else {
    // Agendar retry com backoff exponencial
    const delayMs = 1000 * Math.pow(2, retryCount) // 1s, 2s, 4s, 8s
    const nextAttemptAt = new Date(Date.now() + delayMs)
    
    await db.query(`
      UPDATE workflow_steps 
      SET status = 'FAILED',
          error = $1,
          retry_count = $2,
          next_attempt_at = $3
      WHERE id = $4
    `, [error.message, retryCount, nextAttemptAt, step.id])
  }
}

// Iniciar worker
process.on('SIGTERM', async () => {
  logger.info('Worker encerrando...')
  await db.end()
  process.exit(0)
})

processPendingSteps().catch(console.error)

// api/index.ts - API REST para gerenciar workflows
import express from 'express'
import { Pool } from 'pg'
import { v4 as uuidv4 } from 'uuid'
import cors from 'cors'
import rateLimit from 'express-rate-limit'

const app = express()
const db = new Pool({ connectionString: process.env.DATABASE_URL })

app.use(cors())
app.use(express.json())

// Rate limit: 100 workflows/minuto por IP
const limiter = rateLimit({ windowMs: 60 * 1000, max: 100 })
app.use('/workflows', limiter)

// 1. Criar workflow
app.post('/workflows', async (req, res) => {
  const { tenantId, name, version, definition } = req.body
  
  const workflowId = uuidv4()
  
  await db.query(`
    INSERT INTO workflows (id, tenant_id, name, version, definition, is_active)
    VALUES ($1, $2, $3, $4, $5, $6)
  `, [workflowId, tenantId, name, version, JSON.stringify(definition), true])
  
  res.json({ workflowId, status: 'created' })
})

// 2. Listar workflows
app.get('/workflows/:tenantId', async (req, res) => {
  const { tenantId } = req.params
  
  const result = await db.query(`
    SELECT id, name, version, is_active, created_at 
    FROM workflows 
    WHERE tenant_id = $1 
    ORDER BY created_at DESC
  `, [tenantId])
  
  res.json(result.rows)
})

// 3. Executar workflow (CRÍTICO)
app.post('/workflows/:workflowId/execute', async (req, res) => {
  const { workflowId } = req.params
  const { tenantId, input } = req.body
  
  const executionId = uuidv4()
  
  // Buscar definição do workflow
  const workflow = await db.query(`
    SELECT definition FROM workflows WHERE id = $1 AND is_active = true
  `, [workflowId])
  
  if (workflow.rows.length === 0) {
    return res.status(404).json({ error: 'Workflow não encontrado' })
  }

  const definition = workflow.rows[0].definition
  
  // Criar execução
  await db.query(`
    INSERT INTO workflow_executions (id, workflow_id, tenant_id, input, status)
    VALUES ($1, $2, $3, $4, 'RUNNING')
  `, [executionId, workflowId, tenantId, JSON.stringify(input || {})])

  // Criar steps
  for (let i = 0; i < definition.steps.length; i++) {
    const step = definition.steps[i]
    await db.query(`
      INSERT INTO workflow_steps (
        id, execution_id, step_index, step_id, type, config, status
      ) VALUES ($1, $2, $3, $4, $5, $6, 'PENDING')
    `, [
      uuidv4(),
      executionId,
      i,
      step.stepId,
      step.type,
      JSON.stringify(step)
    ])
  }

  res.json({ executionId, status: 'started' })
})

// 4. Status de execução
app.get('/executions/:executionId', async (req, res) => {
  const { executionId } = req.params
  
  const execution = await db.query(`
    SELECT e.*, w.name as workflow_name
    FROM workflow_executions e
    JOIN workflows w ON e.workflow_id = w.id
    WHERE e.id = $1
  `, [executionId])

  const steps = await db.query(`
    SELECT * FROM workflow_steps 
    WHERE execution_id = $1 
    ORDER BY step_index
  `, [executionId])

  res.json({
    execution: execution.rows[0],
    steps: steps.rows
  })
})

// 5. Dashboard métricas (24h)
app.get('/dashboard/:tenantId', async (req, res) => {
  const { tenantId } = req.params
  const result = await db.query(`
    SELECT 
      w.name,
      COUNT(*) as total,
      SUM(CASE WHEN e.status='COMPLETED' THEN 1 ELSE 0 END) as success,
      ROUND(AVG(EXTRACT(EPOCH FROM (e.finished_at - e.started_at))), 2) as avg_duration
    FROM workflow_executions e
    JOIN workflows w ON e.workflow_id = w.id
    WHERE e.tenant_id = $1 
      AND e.started_at > NOW() - INTERVAL '24 hours'
    GROUP BY w.name
    ORDER BY total DESC
  `, [tenantId])
  
  res.json(result.rows)
})

const PORT = process.env.PORT || 3000
app.listen(PORT, () => {
  console.log(`Workflow API rodando na porta ${PORT}`)
})
