'use client'

import { useState, useEffect, useRef } from 'react'
import mapboxgl, { Map as MapboxMap } from 'mapbox-gl'
import MapboxGeocoder from '@mapbox/mapbox-gl-geocoder'
import { SearchBarSemantic } from '@/components/SearchBarSemantic'
import PropertyCard from '@/components/PropertyCard'
import { supabase } from '@/lib/supabase'
import 'mapbox-gl/dist/mapbox-gl.css'
import '@mapbox/mapbox-gl-geocoder/dist/mapbox-gl-geocoder.css'

mapboxgl.accessToken = process.env.NEXT_PUBLIC_MAPBOX_TOKEN!

export default function ExplorePage() {
  const [properties, setProperties] = useState<any[]>([])
  const [map, setMap] = useState<MapboxMap | null>(null)
  const [selectedProperties, setSelectedProperties] = useState<any[]>([])
  const [userProfile, setUserProfile] = useState<'tenant' | 'investor'>('tenant')
  const mapContainer = useRef<HTMLDivElement>(null)
  const geocoderContainer = useRef<HTMLDivElement>(null)

  // Carregar imóveis iniciais
  useEffect(() => {
    fetchProperties()
  }, [])

  const fetchProperties = async () => {
    const { data } = await supabase
      .from('properties')
      .select('*')
      .limit(100)
    setProperties(data || [])
  }

  // Inicializar Mapbox
  useEffect(() => {
    if (!mapContainer.current) return

    const mapInstance = new mapboxgl.Map({
      container: mapContainer.current,
      style: 'mapbox://styles/mapbox/streets-v12',
      center: [-46.6333, -23.5505], // São Paulo
      zoom: 11,
      pitch: 45,
      bearing: -17.6,
    })

    mapInstance.on('load', () => {
      setMap(mapInstance)
      
      // Fonte de dados
      mapInstance.addSource('properties', {
        type: 'geojson',
        data: {
          type: 'FeatureCollection',
          features: properties.map(p => ({
            type: 'Feature',
            geometry: {
              type: 'Point',
              coordinates: [p.lng, p.lat]
            },
            properties: {
              id: p.id,
              title: p.title,
              price: p.price,
              rating: p.rating,
              roi: p.roi
            }
          }))
        },
        cluster: true,
        clusterMaxZoom: 14,
        clusterRadius: 50
      })

      // Clusters
      mapInstance.addLayer({
        id: 'clusters',
        type: 'circle',
        source: 'properties',
        filter: ['has', ['get', 'point_count']],
        paint: {
          'circle-color': [
            'step',
            ['get', 'point_count'],
            '#51bbd6', // Azul claro
            100,
            '#f1f075', // Amarelo
            750,
            '#f28cb1' // Rosa
          ],
          'circle-radius': [
            'step',
            ['get', 'point_count'],
            20,
            100,
            30,
            750,
            40
          ]
        }
      })

      // Números nos clusters
      mapInstance.addLayer({
        id: 'cluster-count',
        type: 'symbol',
        source: 'properties',
        filter: ['has', ['get', 'point_count']],
        layout: {
          'text-field': '{point_count_abbreviated}',
          'text-font': ['DIN Offc Pro Medium', 'Arial Unicode MS Bold'],
          'text-size': 12
        }
      })

      // Pins individuais
      mapInstance.addLayer({
        id: 'unclustered-point',
        type: 'circle',
        source: 'properties',
        filter: ['!', ['has', ['get', 'point_count']]],
        paint: {
          'circle-color': [
            'case',
            ['>=', ['get', 'roi'], 8], '#00f5d4', // Verde ROI alto
            ['>=', ['get', 'price'], 800], '#ff6b6b', // Vermelho caro
            '#4ade80' // Verde padrão
          ],
          'circle-radius': 12,
          'circle-stroke-width': 2,
          'circle-stroke-color': '#fff'
        }
      })

      // Geocoder
      if (geocoderContainer.current && !mapInstance.getSource('geocoder')) {
        const geocoder = new MapboxGeocoder({
          accessToken: mapboxgl.accessToken!,
          mapboxgl: mapboxgl,
          placeholder: 'Digite endereço ou bairro...'
        })
        geocoderContainer.current.appendChild(geocoder.onAdd(mapInstance))
      }

      // Interações
      mapInstance.on('click', 'clusters', (e) => {
        const features = mapInstance.queryRenderedFeatures(e.point, {
          layers: ['clusters']
        })
        const clusterId = features[0].properties!.cluster_id
        mapInstance.getSource('properties')!.getClusterExpansionZoom(
          clusterId!,
          (err, zoom) => {
            if (zoom) {
              mapInstance.easeTo({
                center: features[0].geometry.coordinates as [number, number],
                zoom: zoom
              })
            }
          }
        )
      })

      mapInstance.on('click', 'unclustered-point', async (e) => {
        const coordinates = e.features![0].geometry.coordinates.slice() as [number, number]
        const propertyId = e.features![0].properties!.id
        
        const { data: property } = await supabase
          .from('properties')
          .select('*')
          .eq('id', propertyId)
          .single()

        setSelectedProperties([property])
        new mapboxgl.Popup()
          .setLngLat(coordinates)
          .setHTML(`<h3>${property?.title}</h3><p>R$ ${property?.price}/noite</p>`)
          .addTo(mapInstance)
      })

      mapInstance.on('mouseenter', 'clusters', () => {
        mapInstance.getCanvas().style.cursor = 'pointer'
      })
      mapInstance.on('mouseleave', 'clusters', () => {
        mapInstance.getCanvas().style.cursor = ''
      })
    })

    return () => mapInstance.remove()
  }, [properties])

  const handleSearchResults = (results: any[]) => {
    setSelectedProperties(results)
    if (map && results.length > 0) {
      map.flyTo({
        center: [results[0].lng, results[0].lat],
        zoom: 14
      })
    }
  }

  return (
    <div className="min-h-screen bg-gradient-to-br from-neutral-50 via-white to-emerald-50">
      {/* Hero */}
      <div className="relative bg-gradient-to-r from-emerald-600 via-emerald-500 to-blue-600 text-white py-20 px-6 overflow-hidden">
        <div className="absolute inset-0 bg-black/10" />
        <div className="relative max-w-7xl mx-auto text-center z-10">
          <h1 className="text-5xl md:text-7xl font-black mb-8 leading-tight">
            Descubra imóveis <span className="bg-white/20 px-8 py-4 rounded-3xl inline-block">perfeitos</span>
          </h1>
          <p className="text-xl md:text-2xl opacity-90 mb-12 max-w-3xl mx-auto leading-relaxed">
            IA semântica + Mapbox clustering = encontre seu imóvel em 3 cliques
          </p>
          <SearchBarSemantic onResults={handleSearchResults} />
        </div>
      </div>

      <div className="max-w-7xl mx-auto px-6 -mt-20 relative z-10">
        <div className="grid grid-cols-1 lg:grid-cols-4 gap-8">
          {/* Mapa Principal */}
          <div className="lg:col-span-3 rounded-3xl overflow-hidden shadow-2xl bg-white/80 backdrop-blur-xl border border-white/50">
            <div ref={mapContainer} className="h-[600px] w-full" />
            <div ref={geocoderContainer} className="absolute top-4 left-4 z-20 bg-white/90 backdrop-blur-xl rounded-2xl shadow-2xl p-2 border" />
          </div>

          {/* Properties Selecionadas */}
          <div className="space-y-6">
            <div className="bg-white/80 backdrop-blur-xl rounded-3xl p-6 shadow-2xl border border-white/50 sticky top-6">
              <h3 className="font-bold text-xl mb-4 flex items-center gap-2">
                <span className="w-2 h-2 bg-emerald-500 rounded-full" />
                {selectedProperties.length || properties.length} imóveis encontrados
              </h3>
              <div className="space-y-4 max-h-96 overflow-y-auto">
                {selectedProperties.length > 0 
                  ? selectedProperties.map((property) => (
                      <PropertyCard
                        key={property.id}
                        property={property}
                        userProfile={userProfile}
                        onReserve={(id) => console.log('Reserve:', id)}
                      />
                    ))
                  : properties.slice(0, 3).map((property) => (
                      <PropertyCard
                        key={property.id}
                        property={property}
                        userProfile={userProfile}
                        onReserve={(id) => console.log('Reserve:', id)}
                      />
                    ))
                }
              </div>
            </div>
          </div>
        </div>
      </div>
    </div>
  )
}
