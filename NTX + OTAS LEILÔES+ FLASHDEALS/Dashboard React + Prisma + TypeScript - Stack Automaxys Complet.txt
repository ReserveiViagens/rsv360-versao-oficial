  tenantId    String             // RSV360 multi-tenant
  name        String
  version     String
  isActive    Boolean            @default(true)
  definition  Json
  createdAt   DateTime           @default(now())
  updatedAt   DateTime           @updatedAt

  @@map("workflows")
}

model WorkflowExecution {
  id            String              @id @default(cuid())
  workflowId    String
  tenantId      String
  status        ExecutionStatus     @default(RUNNING)
  input         Json?
  output        Json?
  currentStep   Int                 @default(0)
  avgDuration   Float?              // Cache para dashboard
  startedAt     DateTime            @default(now())
  finishedAt    DateTime?
  
  steps         WorkflowStep[]
  checkpoints   WorkflowCheckpoint[]
  
  workflow      Workflow            @relation(fields: [workflowId], references: [id])
  
  @@map("workflow_executions")
}

model WorkflowStep {
  id           String     @id @default(cuid())
  executionId  String
  stepIndex    Int
  stepId       String
  type         StepType
  status       StepStatus @default(PENDING)
  input        Json?
  output       Json?
  error        String?
  retryCount   Int        @default(0)
  
  execution    WorkflowExecution @relation(fields: [executionId], references: [id])
  
  @@unique([executionId, stepIndex])
  @@map("workflow_steps")
}

enum ExecutionStatus {
  RUNNING COMPLETED FAILED TIMEOUT CANCELLED
}

enum StepType {
  HTTP QUEUE DELAY CONDITION INTERNAL WEBHOOK
}

enum StepStatus {
  PENDING RUNNING COMPLETED FAILED RETRYING SKIPPED
}

// app/api/dashboard/[tenantId]/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { prisma } from '@/lib/prisma'

export async function GET(
  request: NextRequest,
  { params }: { params: { tenantId: string } }
) {
  const { tenantId } = params
  
  try {
    const metrics = await prisma.$queryRaw<WorkflowMetrics[]>`
      SELECT 
        w.name,
        COUNT(*)::float as total,
        SUM(CASE WHEN e.status='COMPLETED' THEN 1 ELSE 0 END)::float as success,
        ROUND(AVG(EXTRACT(EPOCH FROM (e.finished_at - e.started_at)))::numeric, 2) as avg_duration
      FROM workflow_executions e
      JOIN workflows w ON e.workflow_id = w.id
      WHERE e.tenant_id = ${tenantId} 
        AND e.started_at > NOW() - INTERVAL '24 hours'
      GROUP BY w.id, w.name
      ORDER BY total DESC
    `

    const formattedMetrics = metrics.map(m => ({
      ...m,
      successRate: Number(m.success) / Number(m.total),
      failureRate: 1 - (Number(m.success) / Number(m.total))
    }))

    return NextResponse.json(formattedMetrics)
  } catch (error) {
    return NextResponse.json({ error: 'Erro ao buscar métricas' }, { status: 500 })
  }
}

// middleware.ts
import { withAuth } from 'next-auth/middleware'
import { NextResponse } from 'next/server'

export default withAuth(
  function middleware(req) {
    const tenantId = req.nextauth.token?.tenantId
    if (!tenantId) {
      return NextResponse.redirect(new URL('/unauthorized', req.url))
    }
    
    // Headers para tenant isolation
    const requestHeaders = new Headers(req.headers)
    requestHeaders.set('x-tenant-id', tenantId as string)
    
    return NextResponse.next({
      request: {
        headers: requestHeaders,
      },
    })
  },
  {
    callbacks: {
      authorized: ({ token }) => !!token
    },
  }
)

export const config = {
  matcher: ['/dashboard/:path*']

// app/api/dashboard/[tenantId]/route.ts
import { NextRequest } from 'next/server'
import { prisma } from '@/lib/prisma'

export async function GET(req: NextRequest) {
  const tenantId = req.headers.get('x-tenant-id') || req.nextUrl.searchParams.get('tenantId')
  
  if (!tenantId) {
    return Response.json({ error: 'Tenant ID required' }, { status: 400 })
  }

  const metrics = await prisma.$queryRaw`
    SELECT 
      w.name,
      COUNT(e.id)::float as total,
      SUM(CASE WHEN e.status='COMPLETED' THEN 1 ELSE 0 END)::float as success,
      ROUND(AVG(EXTRACT(EPOCH FROM COALESCE(e.finished_at, NOW()) - e.started_at)*1000)::numeric, 0) as avg_duration_ms,
      PERCENTILE_CONT(0.95) WITHIN GROUP (ORDER BY EXTRACT(EPOCH FROM COALESCE(e.finished_at, NOW()) - e.startedAt)*1000) as p95_duration_ms
    FROM workflow_executions e
    JOIN workflows w ON e.workflow_id = w.id
    WHERE e.tenant_id = ${tenantId}::text
      AND e.started_at > NOW() - INTERVAL '24 hours'
    GROUP BY w.id, w.name
    ORDER BY total DESC
  `

  return Response.json(metrics)
}

// components/dashboard/MetricsGrid.tsx
'use client'

import { useEffect, useState } from 'react'
import { Card, Statistic, Progress, Row, Col } from 'antd'
import { RefreshCw, TrendingUp, AlertCircle } from 'lucide-react'

interface MetricsData {
  total: number
  success: number
  successRate: number
  avgDuration: number
}

export default function MetricsGrid({ tenantId }: { tenantId: string }) {
  const [metrics, setMetrics] = useState<MetricsData>({
    total: 0, success: 0, successRate: 0, avgDuration: 0
  })
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    fetchMetrics()
    const interval = setInterval(fetchMetrics, 5000)
    return () => clearInterval(interval)
  }, [tenantId])

  const fetchMetrics = async () => {
    try {
      const res = await fetch(`/api/dashboard/${tenantId}`)
      const data = await res.json()
      
      const total = data.reduce((sum: number, m: any) => sum + m.total, 0)
      const success = data.reduce((sum: number, m: any) => sum + m.success, 0)
      
      setMetrics({
        total,
        success,
        successRate: total > 0 ? success / total : 0,
        avgDuration: data.reduce((sum: number, m: any) => sum + (m.avgDurationMs || 0) * m.total, 0) / total
      })
      setLoading(false)
    } catch (error) {
      console.error('Erro métricas:', error)
    }
  }

  return (
    <Row gutter={[24, 24]} className="mb-8">
      <Col xs={24} sm={12} lg={6}>
        <Card>
          <Statistic
            title="Total Execuções"
            value={metrics.total}
            prefix={<RefreshCw className="w-4 h-4" />}
            valueStyle={{ color: '#1677ff' }}
          />
        </Card>
      </Col>
      <Col xs={24} sm={12} lg={6}>
        <Card>
          <Statistic
            title="Taxa Sucesso"
            value={Math.round(metrics.successRate * 100)}
            suffix="%" 
            valueStyle={{ color: metrics.successRate > 0.95 ? '#52c41a' : '#fa8c16' }}
          >
            <Progress 
              percent={metrics.successRate * 100} 
              size={40} 
              showInfo={false}
              strokeColor={metrics.successRate > 0.95 ? '#52c41a' : '#fa8c16'}
            />
          </Statistic>
        </Card>
      </Col>
      <Col xs={24} sm={12} lg={6}>
        <Card>
          <Statistic
            title="Duração Média"
            value={Math.round(metrics.avgDuration)}
            suffix="ms"
            prefix={<TrendingUp className="w-4 h-4" />}
          />
        </Card>
      </Col>
      <Col xs={24} sm={12} lg={6}>
        <Card>
          <Statistic
            title="Falhas 24h"
            value={metrics.total - metrics.success}
            valueStyle={{ color: '#ff4d4f' }}
            prefix={<AlertCircle className="w-4 h-4" />}
          />
        </Card>
      </Col>
    </Row>
  )
}

// middleware/ota-auth.ts
export async function validateOTAAuth(req: Request) {
  const signature = req.headers.get('x-signature')
  const apiKey = req.headers.get('x-api-key')
  const payload = await req.text()
  
  // 1. Validar HMAC signature
  const expectedSignature = crypto
    .createHmac('sha256', apiKey)
    .update(payload)
    .digest('hex')
    
  if (signature !== expectedSignature) {
    throw new Error('Invalid webhook signature')
  }
  
  // 2. Validar tenant + OTA
  const data = JSON.parse(payload)
  const otaConfig = await prisma.otaConfig.findFirst({
    where: { 
      hotelId: data.hotelId, 
      ota: data.ota,
      apiKey 
    }
  })
  
  if (!otaConfig) throw new Error('OTA not authorized')
  
  return { tenantId: data.hotelId, otaConfig }
}

// services/ota-idempotency.ts
export async function checkDuplicate(reservation: OTAReservation) {
  // 1. Verificar cache Redis (TTL 24h)
  const cacheKey = `ota_dup:${reservation.ota}:${reservation.reservationId}`
  const cached = await redis.get(cacheKey)
  if (cached) return true
  
  // 2. Verificar RSV360 reservations
  const exists = await prisma.reservation.findFirst({
    where: {
      externalId: reservation.reservationId,
      source: reservation.ota,
      hotelId: reservation.hotelId
    }
  })
  
  if (exists) {
    await redis.setex(cacheKey, 86400, '1') // 24h TTL
    return true
  }
  
  return false
}
Circuit Breaker OTA
typescript
// services/ota-circuit-breaker.ts
class OTACircuitBreaker {
  private consecutiveErrors = 0
  private state: 'CLOSED' | 'OPEN' | 'HALF_OPEN' = 'CLOSED'
  private lastFailure: Date | null = null
  
  async execute<T>(ota: string, fn: () => Promise<T>): Promise<T> {
    if (this.state === 'OPEN') {
      const timeout = Date.now() - (this.lastFailure?.getTime() || 0)
      if (timeout < 30000) throw new Error(`OTA ${ota} circuit open`)
      this.state = 'HALF_OPEN'
    }
    
    try {
      const result = await fn()
      this.consecutiveErrors = 0
      this.state = 'CLOSED'
      return result
    } catch (error) {
      this.consecutiveErrors++
      this.lastFailure = new Date()
      
      if (this.consecutiveErrors >= 5) {
        this.state = 'OPEN'
      }
      
      throw error
    }
  }
}
