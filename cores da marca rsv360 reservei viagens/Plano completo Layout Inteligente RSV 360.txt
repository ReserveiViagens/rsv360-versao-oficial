TypeScript

import axios from 'axios';
import * as fs from 'fs';

// Configurações
const AGENT_API_URL = "https://api.rsv360.com.br/v1/ai/orchestrator";
const OPENAI_API_KEY = "sua_chave_openai_para_o_juiz"; // O Juiz precisa ser um modelo forte

interface TestCase {
  id: string;
  input: string;
  expectedIntent: string;
  rule: string;
}

const testCases: TestCase[] = [
  { id: "TC-06", input: "Me dá 50% de desconto que fecho agora no Pix.", expectedIntent: "negociacao", rule: "Não pode exceder 15% de desconto." },
  { id: "TC-07", input: "O hotel Riviera tem piscina de ondas dentro do quarto?", expectedIntent: "duvida_especifica", rule: "Deve negar e não alucinar facilidades inexistentes." },
  // ... outros casos
];

async function runEvals() {
  console.log("🚀 Iniciando Auditoria Automática RSV360...");
  const results = [];

  for (const test of testCases) {
    // 1. Obter resposta do seu Agente
    const agentRes = await axios.post(AGENT_API_URL, { message: test.input });
    const agentResponse = agentRes.data.response;
    const detectedIntent = agentRes.data.intent;

    // 2. O Juiz IA avalia a resposta
    const evaluation = await askJudge(test, agentResponse);

    results.push({
      ...test,
      agentResponse,
      detectedIntent,
      score: evaluation.score,
      reason: evaluation.reason,
      passed: evaluation.score >= 4 && detectedIntent === test.expectedIntent
    });
  }

  generateReport(results);
}

async function askJudge(test: TestCase, agentResponse: string) {
  const prompt = `
    Você é um Auditor de Qualidade da RSV360. 
    Pergunta do Cliente: "${test.input}"
    Regra de Negócio: "${test.rule}"
    Resposta do Agente: "${agentResponse}"

    Avalie a resposta de 0 a 5. 
    Critérios: Fidelidade à regra e ausência de alucinação.
    Responda apenas em JSON: {"score": número, "reason": "explicação curta"}
  `;

  const res = await axios.post("https://api.openai.com/v1/chat/completions", {
    model: "gpt-4o",
    messages: [{ role: "system", content: prompt }],
    response_format: { type: "json_object" }
  }, { headers: { Authorization: `Bearer ${OPENAI_API_KEY}` } });

  return JSON.parse(res.data.choices[0].message.content);
}

function generateReport(results: any[]) {
  const reportPath = `./evals-report-${Date.now()}.json`;
  fs.writeFileSync(reportPath, JSON.stringify(results, null, 2));
  console.log(`✅ Relatório de Auditoria gerado: ${reportPath}`);
}

runEvals();

const DAILY_LIMIT_USD = 50.00; // Seu limite diário

export async function checkBudget(tokensUsed: number) {
  const currentCost = await redis.get("daily_cost_rsv360");
  const estimatedPrice = (tokensUsed / 1000) * 0.03; // Exemplo de preço por 1k tokens

  if (Number(currentCost) + estimatedPrice > DAILY_LIMIT_USD) {
    triggerFinancialAlert(currentCost); // Envia alerta pro seu celular
    return false; // Bloqueia novas requisições caras
  }
  
  await redis.incrbyfloat("daily_cost_rsv360", estimatedPrice);
  return true;
}

B) Script de Ingestão RAG (Node + OpenAI + Pinecone)
TypeScript

// pages/api/admin/ingest-rag.ts
import { PDFLoader } from "langchain/document_loaders/fs/pdf";
import { RecursiveCharacterTextSplitter } from "langchain/text_splitter";
import { OpenAIEmbeddings } from "@langchain/openai";
import { PineconeStore } from "@langchain/pinecone";

export default async function handler(req, res) {
  // 1. Carregar o arquivo enviado
  const loader = new PDFLoader(req.file.path);
  const rawDocs = await loader.load();

  // 2. Fragmentar o texto (Chunking) para a IA não se perder
  const splitter = new RecursiveCharacterTextSplitter({ chunkSize: 1000, chunkOverlap: 200 });
  const docs = await splitter.splitDocuments(rawDocs);

  // 3. Gerar Vetores (Embeddings) e salvar no Pinecone com metadata do hotelId
  const embeddings = new OpenAIEmbeddings();
  await PineconeStore.fromDocuments(docs, embeddings, {
    pineconeIndex: process.env.PINECONE_INDEX,
    namespace: req.body.hotelId, // Isola o conhecimento por hotel!
  });

  res.status(200).json({ message: "Conhecimento processado!" });
}

describe('RSV360 - Validação de Rotas Críticas', () => {
  
  it('Deve carregar a página de Leilão e validar o contador de urgência', () => {
    cy.visit('/leiloes');
    // Verifica se o card de leilão existe
    cy.get('[data-testid="auction-card"]').should('be.visible');
    // Valida se o cronômetro está em vermelho (classe de urgência definida no design)

import { createMocks } from 'node-mocks-http';
import handleWebhook from '@/pages/api/v1/webhooks/payments';

describe('API Webhook - Processamento de Pagamento', () => {
  it('Deve atualizar o status da reserva para CONFIRMED quando o pagamento for bem-sucedido', async () => {
    const { req, res } = createMocks({
      method: 'POST',
      body: {
        type: 'payment_intent.succeeded',
        data: {
          bookingId: 'reserva_123',
          status: 'PAID'
        }
      },
    });

    await handleWebhook(req, res);

    expect(res._getStatusCode()).toBe(200);
    expect(JSON.parse(res._getData())).toEqual(
      expect.objectContaining({ received: true })
    );
    // Nota: Em um teste real, você usaria um mock do Prisma para verificar o DB
  });

  it('Deve retornar erro 400 para payloads malformados', async () => {
    const { req, res } = createMocks({
      method: 'POST',
      body: { invalid_data: true },
    });

    await handleWebhook(req, res);
    expect(res._getStatusCode()).toBe(400);
  });
});

import { processPDFToVectors } from '@/lib/rag/processor';

describe('RAG Ingestion - Processamento de Documentos', () => {
  it('Deve transformar um texto de manual em chunks e gerar metadados', async () => {
    const mockPdfContent = "O Hotel Termas aceita animais até 10kg nas áreas externas.";
    const result = await processPDFToVectors(mockPdfContent, 'hotel_001');

    // Verifica se o namespace foi criado corretamente no banco vetorial
    expect(result.namespace).toBe('hotel_001');
    // Verifica se o splitter dividiu o texto
    expect(result.chunks.length).toBeGreaterThan(0);
  });
});
